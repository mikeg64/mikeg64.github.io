535 
536 #ifdef FEEDBACK
537  for (i=is; i<=ie; i++) {
538  pG->U[ks][js][i].M1 -= pG->Coup[ks][js][i].fb1;
539  pG->U[ks][js][i].M2 -= pG->Coup[ks][js][i].fb2;
540  pG->U[ks][js][i].M3 -= pG->Coup[ks][js][i].fb3;
541 #ifndef BAROTROPIC
542  pG->U[ks][js][i].E += pG->Coup[ks][js][i].Eloss;
543  pG->Coup[ks][js][i].Eloss *= dt1;
544 #endif
545  }
546 #endif
547 
548 /*=== STEP 12: Update cell-centered values for a full timestep ===============*/
549 
550 /*--- Step 12a -----------------------------------------------------------------
551  * Update cell-centered variables in pG using 1D x1-fluxes
552  */
553 
554  for (i=is; i<=ie; i++) {
555 #ifdef CYLINDRICAL
556  rsf = ri[i+1]/r[i]; lsf = ri[i]/r[i];
557 #endif
558  pG->U[ks][js][i].d -= dtodx1*(rsf*x1Flux[i+1].d - lsf*x1Flux[i].d );
559  pG->U[ks][js][i].M1 -= dtodx1*(rsf*x1Flux[i+1].Mx - lsf*x1Flux[i].Mx);
560  pG->U[ks][js][i].M2 -= dtodx1*(SQR(rsf)*x1Flux[i+1].My - SQR(lsf)*x1Flux[i].My);
561  pG->U[ks][js][i].M3 -= dtodx1*(rsf*x1Flux[i+1].Mz - lsf*x1Flux[i].Mz);
562 #ifndef BAROTROPIC
563  pG->U[ks][js][i].E -= dtodx1*(rsf*x1Flux[i+1].E - lsf*x1Flux[i].E );
564 #endif /* BAROTROPIC */
565 #ifdef MHD
566  pG->U[ks][js][i].B2c -= dtodx1*(x1Flux[i+1].By - x1Flux[i].By);
567  pG->U[ks][js][i].B3c -= dtodx1*(rsf*x1Flux[i+1].Bz - lsf*x1Flux[i].Bz);
568 /* For consistency, set B2i and B3i to cell-centered values. */
569  pG->B2i[ks][js][i] = pG->U[ks][js][i].B2c;
570  pG->B3i[ks][js][i] = pG->U[ks][js][i].B3c;
571 #endif /* MHD */
572 #if (NSCALARS > 0)
573  for (n=0; n<NSCALARS; n++)
574  pG->U[ks][js][i].s[n] -= dtodx1*(rsf*x1Flux[i+1].s[n] - lsf*x1Flux[i].s[n]);
575 #endif
576  }
577 
578 /*--- Step 12b: Not needed in 1D ---*/
579 /*--- Step 12c: Not needed in 1D ---*/
580 /*--- Step 12d: Not needed in 1D ---*/

585  */
586

597  pG->CGrid[ncg].myFlx[dim][ks][js].M2 = x1Flux[i].My;
598  pG->CGrid[ncg].myFlx[dim][ks][js].M3 = x1Flux[i].Mz;
599 #ifndef BAROTROPIC
600  pG->CGrid[ncg].myFlx[dim][ks][js].E = x1Flux[i].E;
601 #endif /* BAROTROPIC */
602 #ifdef MHD
603  pG->CGrid[ncg].myFlx[dim][ks][js].B1c = 0.0;
604  pG->CGrid[ncg].myFlx[dim][ks][js].B2c = x1Flux[i].By;
605  pG->CGrid[ncg].myFlx[dim][ks][js].B3c = x1Flux[i].Bz;
606 #endif /* MHD */
607 #if (NSCALARS > 0)
608  for (n=0; n<NSCALARS; n++)
609  pG->CGrid[ncg].myFlx[dim][ks][js].s[n] = x1Flux[i].s[n];
610 #endif
611  }
612  }
613  }
614 
615 /* x1-boundaries of parent Grids (at boundaries of THIS Grid) */
616  for (npg=0; npg<pG->NPGrid; npg++) {
617  for (dim=0; dim<2; dim++){
618  if (pG->PGrid[npg].myFlx[dim] != NULL) {
619

621  if (dim==1) i = pG->PGrid[npg].ijke[0] + 1;
622

624  pG->PGrid[npg].myFlx[dim][ks][js].M1 = x1Flux[i].Mx;
625  pG->PGrid[npg].myFlx[dim][ks][js].M2 = x1Flux[i].My;
626  pG->PGrid[npg].myFlx[dim][ks][js].M3 = x1Flux[i].Mz;
627 #ifndef BAROTROPIC
628  pG->PGrid[npg].myFlx[dim][ks][js].E = x1Flux[i].E;
629 #endif /* BAROTROPIC */
630 #ifdef MHD
631  pG->PGrid[npg].myFlx[dim][ks][js].B1c = 0.0;
632  pG->PGrid[npg].myFlx[dim][ks][js].B2c = x1Flux[i].By;
633  pG->PGrid[npg].myFlx[dim][ks][js].B3c = x1Flux[i].Bz;
634 #endif /* MHD */

643 #endif /* STATIC_MESH_REFINEMENT */
