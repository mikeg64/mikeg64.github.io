55  js = pG->js; je = pG->je; nx2 = je-js+1;
56  ks = pG->ks; ke = pG->ke; nx3 = ke-ks+1;

61 
62 #ifndef CYLINDRICAL

64 #endif
65 
66  if (nx1==1) {
67  ath_error("[cyladvect]: This problem can only be run in 2D or 3D!\n");
68  }
69  else if (nx2==1 && nx3>1) {
70  ath_error("[cyladvect]: Only (R,phi) can be used in 2D!\n");
71  }

73  /* allocate memory for solution */

92  /* Period is one grid-crossing */
93  alpha = 2.0*PI/(x2max-x2min);
94 
95  /* Set density and phi-momentum */
96  for (k=kl; k<=ku; k++) {
97  for (j=jl; j<=ju; j++) {
98  for (i=il; i<=iu; i++) {
99  cc_pos(pG,i,j,k,&x1,&x2,&x3);
100  memset(&(pG->U[k][j][i]),0.0,sizeof(ConsS));
101 
102  switch (iprob) {
103  case 1:
104  mask = (int)((x1>=R0-rad) && (x1<=R0+rad) && (x2>=phi0-rad/R0) && (x2<=phi0+rad/R0));
105  pG->U[k][j][i].d = rho0*(1.0 + amp*mask);
106 #ifndef FARGO

108 #else

118 #ifndef FARGO
119  pG->U[k][j][i].M2 = pG->U[k][j][i].d*x1*omega0;
120 #else
121  pG->U[k][j][i].M2 = 0;
122 #endif
123  break;
124  case 3:
125  x = x1*cos(x2); y = x1*sin(x2); z = x3;
126  x0 = R0*cos(phi0); y0 = R0*sin(phi0);
127  r = sqrt(SQR(x-x0) + SQR(y-y0) + SQR(z-z0));
128  pG->U[k][j][i].d = rho0*(1.0 + exp(-0.5*SQR(3.0*r/rad)));
129 #ifndef FARGO
130  pG->U[k][j][i].M2 = pG->U[k][j][i].d*x1*omega0;
131 #else
132  pG->U[k][j][i].M2 = 0;
133 #endif
134 #ifdef MHD
135  pG->U[k][j][i].B3c = bz0;
136  pG->B3i[k][j][i] = bz0;
137 #endif
138  break;
139  default:
140  ath_error("[cyladvect]: Not an accepted problem number\n");
141  }
142 
143  pG->U[k][j][i].M3 = pG->U[k][j][i].d*vz0;
144 
145 #ifndef ISOTHERMAL
146  /* Initialize total energy */
147  Emag = 0.0;
148 #ifdef MHD
149  Emag = 0.5*SQR(pG->U[k][j][i].B3c);
150 #endif
151  Eint = (Pgas0-Emag)/Gamma_1;
152  Ekin = 0.5*(SQR(pG->U[k][j][i].M1) + SQR(pG->U[k][j][i].M2) + SQR(pG->U[k][j][i].M3))/pG->U[k][j][i].d;
153 
154  pG->U[k][j][i].E = Eint + Emag + Ekin;
155 #endif /* ISOTHERMAL */
156 
157  /* Save solution */
158  RootSoln[k][j][i] = pG->U[k][j][i];
159  }
160  }
161  }
162 
163  StaticGravPot = grav_pot;
164  if (iprob==2) {
165  bvals_mhd_fun(pDomain, left_x1, cyladvect_ix1);
166  bvals_mhd_fun(pDomain, right_x1, cyladvect_ox1);
167  } else {
168  bvals_mhd_fun(pDomain, left_x1, do_nothing_bc);
169  bvals_mhd_fun(pDomain, right_x1, do_nothing_bc);
170  }
171 #ifdef FARGO
172  OrbitalProfile = Omega;
173  ShearProfile = Shear;
174 #endif
175
