636  * ran2: extracted from the Numerical Recipes in C (version 2) code. Modified
637  * to use doubles instead of floats. -- T. A. Gardiner -- Aug. 12, 2003
638  */
639 
640 #define IM1 2147483563
641 #define IM2 2147483399
642 #define AM (1.0/IM1)
643 #define IMM1 (IM1-1)
644 #define IA1 40014
645 #define IA2 40692
646 #define IQ1 53668
647 #define IQ2 52774
648 #define IR1 12211
649 #define IR2 3791
650 #define NTAB 32
651 #define NDIV (1+IMM1/NTAB)
652 #define RNMX (1.0-DBL_EPSILON)
653 
654 /* Long period (> 2 x 10^{18}) random number generator of L'Ecuyer
655  * with Bays-Durham shuffle and added safeguards. Returns a uniform
656  * random deviate between 0.0 and 1.0 (exclusive of the endpoint
657  * values). Call with idum = a negative integer to initialize;
658  * thereafter, do not alter idum between successive deviates in a
659  * sequence. RNMX should appriximate the largest floating point value
660  * that is less than 1.
661  */
662 
663 double ran2(long int *idum)
664 {
665  int j;
666  long int k;
667  static long int idum2=123456789;
668  static long int iy=0;
669  static long int iv[NTAB];
670  double temp;
671 
672  if (*idum <= 0) { /* Initialize */
673  if (-(*idum) < 1) *idum=1; /* Be sure to prevent idum = 0 */
674  else *idum = -(*idum);
675  idum2=(*idum);
676  for (j=NTAB+7;j>=0;j--) { /* Load the shuffle table (after 8 warm-ups) */
677  k=(*idum)/IQ1;
678  *idum=IA1*(*idum-k*IQ1)-k*IR1;
679  if (*idum < 0) *idum += IM1;
680  if (j < NTAB) iv[j] = *idum;
681  }
682  iy=iv[0];
683  }
684  k=(*idum)/IQ1; /* Start here when not initializing */
685  *idum=IA1*(*idum-k*IQ1)-k*IR1; /* Compute idum=(IA1*idum) % IM1 without */
686  if (*idum < 0) *idum += IM1; /* overflows by Schrage's method */
687  k=idum2/IQ2;
688  idum2=IA2*(idum2-k*IQ2)-k*IR2; /* Compute idum2=(IA2*idum) % IM2 likewise */
689  if (idum2 < 0) idum2 += IM2;
690  j=(int)(iy/NDIV); /* Will be in the range 0...NTAB-1 */
691  iy=iv[j]-idum2; /* Here idum is shuffled, idum and idum2 */
692  iv[j] = *idum; /* are combined to generate output */
693  if (iy < 1) iy += IMM1;
694  if ((temp=AM*iy) > RNMX) return RNMX; /* No endpoint values */
695  else return temp;
696 }
