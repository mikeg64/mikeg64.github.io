571 void Userwork_after_loop(MeshS *pM)
572 {
573  GridS *pGrid;
574  int i=0,j=0,k=0;
575  int is,ie,js,je,ks,ke;
576  Real rms_error=0.0;
577  ConsS error,total_error;
578  FILE *fp;
579  char *fname;
580  int Nx1, Nx2, Nx3, count, min_zones;
581 #if defined MPI_PARALLEL
582  double err[8], tot_err[8];
583  int ierr,myID;
584 #endif
585 
586  int err_test = par_getd_def("problem","error_test",0);
587  if (err_test == 0) return;
588 
589  total_error.d = 0.0;
590  total_error.M1 = 0.0;
591  total_error.M2 = 0.0;
592  total_error.M3 = 0.0;
593 #ifdef MHD
594  total_error.B1c = 0.0;
595  total_error.B2c = 0.0;
596  total_error.B3c = 0.0;
597 #endif /* MHD */
598 #ifndef ISOTHERMAL
599  total_error.E = 0.0;
600 #endif /* ISOTHERMAL */
601 
602 /* Compute error only on root Grid, which is in Domain[0][0] */
603 
604  pGrid=pM->Domain[0][0].Grid;
605  if (pGrid == NULL) return;
606 
607 /* compute L1 error in each variable, and rms total error */
608 
609  is = pGrid->is; ie = pGrid->ie;
610  js = pGrid->js; je = pGrid->je;
611  ks = pGrid->ks; ke = pGrid->ke;
612  for (k=ks; k<=ke; k++) {
613  for (j=js; j<=je; j++) {
614  error.d = 0.0;
615  error.M1 = 0.0;
616  error.M2 = 0.0;
617  error.M3 = 0.0;
618 #ifdef MHD
619  error.B1c = 0.0;
620  error.B2c = 0.0;
621  error.B3c = 0.0;
622 #endif /* MHD */
623 #ifndef ISOTHERMAL
624  error.E = 0.0;
625 #endif /* ISOTHERMAL */
626 
627  for (i=is; i<=ie; i++) {
628  error.d += fabs(pGrid->U[k][j][i].d - RootSoln[k][j][i].d );
629  error.M1 += fabs(pGrid->U[k][j][i].M1 - RootSoln[k][j][i].M1);
630  error.M2 += fabs(pGrid->U[k][j][i].M2 - RootSoln[k][j][i].M2);
631  error.M3 += fabs(pGrid->U[k][j][i].M3 - RootSoln[k][j][i].M3);
632 #ifdef MHD
633  error.B1c += fabs(pGrid->U[k][j][i].B1c - RootSoln[k][j][i].B1c);
634  error.B2c += fabs(pGrid->U[k][j][i].B2c - RootSoln[k][j][i].B2c);
635  error.B3c += fabs(pGrid->U[k][j][i].B3c - RootSoln[k][j][i].B3c);
636 #endif /* MHD */
637 #ifndef ISOTHERMAL
638  error.E += fabs(pGrid->U[k][j][i].E - RootSoln[k][j][i].E );
639 #endif /* ISOTHERMAL */
640  }
641 
642  total_error.d += error.d;
643  total_error.M1 += error.M1;
644  total_error.M2 += error.M2;
645  total_error.M3 += error.M3;
646 #ifdef MHD
647  total_error.B1c += error.B1c;
648  total_error.B2c += error.B2c;
649  total_error.B3c += error.B3c;
650 #endif /* MHD */
651 #ifndef ISOTHERMAL
652  total_error.E += error.E;
653 #endif /* ISOTHERMAL */
654  }}
655 
656 #ifdef MPI_PARALLEL
657  Nx1 = pM->Domain[0][0].Nx[0];
658  Nx2 = pM->Domain[0][0].Nx[1];
659  Nx3 = pM->Domain[0][0].Nx[2];
660 #else
661  Nx1 = ie - is + 1;
662  Nx2 = je - js + 1;
663  Nx3 = ke - ks + 1;
664 #endif
665  count = Nx1*Nx2*Nx3;
666 
667 #ifdef MPI_PARALLEL
668 /* Now we have to use an All_Reduce to get the total error over all the MPI
669  * grids. Begin by copying the error into the err[] array */
670 
671  err[0] = total_error.d;
672  err[1] = total_error.M1;
673  err[2] = total_error.M2;
674  err[3] = total_error.M3;
675 #ifdef MHD
676  err[4] = total_error.B1c;
677  err[5] = total_error.B2c;
678  err[6] = total_error.B3c;
679 #endif /* MHD */
680 #ifndef ISOTHERMAL
681  err[7] = total_error.E;
682 #endif /* ISOTHERMAL */
683 
684  /* Sum up the Computed Error */
685  ierr = MPI_Reduce(err,tot_err,8,MPI_DOUBLE,MPI_SUM,0,
686  pM->Domain[0][0].Comm_Domain);
687 
688 /* If I'm the parent, copy the sum back to the total_error variable */
689 
690  ierr = MPI_Comm_rank(pM->Domain[0][0].Comm_Domain, &myID);
691  if(myID == 0){ /* I'm the parent */
692  total_error.d = tot_err[0];
693  total_error.M1 = tot_err[1];
694  total_error.M2 = tot_err[2];
695  total_error.M3 = tot_err[3];
696 #ifdef MHD
697  total_error.B1c = tot_err[4];
698  total_error.B2c = tot_err[5];
699  total_error.B3c = tot_err[6];
700 #endif /* MHD */
701 #ifndef ISOTHERMAL
702  total_error.E = tot_err[7];
703 #endif /* ISOTHERMAL */
704  }
705  else return; /* The child grids do not do any of the following code */
706 
707 #endif /* MPI_PARALLEL */
708 
709 /* Compute RMS error over all variables, and print out */
710 
711  rms_error = SQR(total_error.d) + SQR(total_error.M1) + SQR(total_error.M2)
712  + SQR(total_error.M3);
713 #ifdef MHD
714  rms_error += SQR(total_error.B1c) + SQR(total_error.B2c)
715  + SQR(total_error.B3c);
716 #endif /* MHD */
717 #ifndef ISOTHERMAL
718  rms_error += SQR(total_error.E);
719 #endif /* ISOTHERMAL */
720  rms_error = sqrt(rms_error)/(double)count;
721 
722 /* Print warning to stdout if rms_error exceeds estimate of 1st-order conv */
723 
724  min_zones = Nx1;
725  if (Nx2 > 1) min_zones = MIN(min_zones,Nx2);
726 #ifdef HYDRO
727  if (rms_error > 8.0/min_zones)
728  printf("WARNING: rms_error=%e exceeds estimate\n",rms_error);
729 #endif
730 #ifdef MHD
731  if (rms_error > 12.0/min_zones)
732  printf("WARNING: rms_error=%e exceeds estimate\n",rms_error);
733 #endif
734 
735 /* Print error to file "LinWave-errors.#.dat", where #=wave_flag */
736 
737 #ifdef MPI_PARALLEL
738  fname = "../shock-errors.dat";
739 #else
740  fname = "shock-errors.dat";
741 #endif
742 
743 /* The file exists -- reopen the file in append mode */
744  if((fp=fopen(fname,"r")) != NULL){
745  if((fp = freopen(fname,"a",fp)) == NULL){
746  ath_perr(-1,"[Userwork_after_loop]: Unable to reopen file.\n");
747  free(fname);
748  return;
749  }
750  }
751 /* The file does not exist -- open the file in write mode */
752  else{
753  if((fp = fopen(fname,"w")) == NULL){
754  ath_perr(-1,"[Userwork_after_loop]: Unable to open file.\n");
755  free(fname);
756  return;
757  }
758 /* Now write out some header information */
759  fprintf(fp,"# Nx1 Nx2 Nx3 RMS-Error d M1 M2 M3");
760 #ifndef ISOTHERMAL
761  fprintf(fp," E");
762 #endif /* ISOTHERMAL */
763 #ifdef MHD
764  fprintf(fp," B1c B2c B3c");
765 #endif /* MHD */
766  fprintf(fp,"\n#\n");
767  }
768 
769  fprintf(fp,"%d %d %d %e",Nx1,Nx2,Nx3,rms_error);
770 
771  fprintf(fp," %e %e %e %e",
772  (total_error.d/(double)count),
773  (total_error.M1/(double)count),
774  (total_error.M2/(double)count),
775  (total_error.M3/(double)count) );
776 
777 #ifndef ISOTHERMAL
778  fprintf(fp," %e",(total_error.E/(double)count) );
779 #endif /* ISOTHERMAL */
780 
781 #ifdef MHD
782  fprintf(fp," %e %e %e",
783  (total_error.B1c/(double)count),
784  (total_error.B2c/(double)count),
785  (total_error.B3c/(double)count));
786 #endif /* MHD */
787 
788  fprintf(fp,"\n");
789 
790  fclose(fp);
791 
792  return;
793 }
