2 /*==============================================================================
3  * FILE: get_eta.c
4  *
5  * PURPOSE: Functions to calculate the diffusion coefficients for resistivity.
6  * There are two standard prescriptions:
7  *
8  * A. Single ion prescription (eq. 21 of Balbus & Terquem 2001, ApJ):
9  * E = eta_Ohm*J + (JxB)/(4pi*n_e*e/c) - [(JxB)xB]/(4pi*gamma*rho_i*rho)
10  *
11  * B. Generalized prescription (eq. 25-31 of Wardle 2007, ApSS):
12  * E = eta_Ohm*J + eta_Hall*(JxB)/B + eta_AD*J_perp
13  * where eta_Ohm, eta_Hall and eta_AD are determined by sigma_O, sigma_H and
14  * sigma_P.
15  *
16  * We provide two options for the user to set the diffusion coefficients:
17  *
18  * CASE 1: eta_Ohm is constant and n_e and/or rho_i are proportional to
19  * some power (d_ind) of the gas density rho. The user needs to provide
20  * eta_Ohm and two proportional coefficients Q_Hall and Q_AD, which are
21  * defined in global.h.
22  *
23  * CASE 2: The user provides his/her own function to evaluate the etas,
24  * e.g., spatially variable diffussivity, or adopting some chemical model,
25  * If the generalized prescription is used, the user can call the function
26  * convert_diffusion to convert the conductivitis to resistivities. The
27  * (blank) function get_eta_user() is given in the problem generator.
28  *
29  * CONTAINS PUBLIC FUNCTIONS:
30  * get_eta() - main function call to get magnetic diffusivities
31  * eta_single_const() - constant diffusivities for single ion prescription
32  * eta_general() - user defined diffusivities
33  * convert_diffusion() - convert conductivities to diffusion coefficients
34  *============================================================================*/

96 void eta_standard(GridS *pG, int i, int j, int k,
97  Real *eta_O, Real *eta_H, Real *eta_A)
98 {
99  Real Bsq, Bmag;
100 
101  *eta_O = eta_Ohm;
102 
103  if ((Q_Hall > 0.0) || (Q_AD > 0.0))
104  {
105  Bsq = SQR(pG->U[k][j][i].B1c) + SQR(pG->U[k][j][i].B2c)
106  + SQR(pG->U[k][j][i].B3c);
107  Bmag = sqrt(Bsq);
108 
109  *eta_H = Q_Hall * Bmag / pow(pG->U[k][j][i].d, d_ind);
110  *eta_A = Q_AD * Bsq / pow(pG->U[k][j][i].d, 1.0+d_ind);
111  }
