606 #endif

610  total_error.M1 += error.M1;
611  total_error.M2 += error.M2;
612  total_error.M3 += error.M3;
613 #ifdef MHD
614  total_error.B1c += error.B1c;
615  total_error.B2c += error.B2c;
616  total_error.B3c += error.B3c;
617 #endif /* MHD */

619  total_error.E += error.E;
620 #endif /* ISOTHERMAL */
621 #if (NSCALARS > 0)
622  for (n=0; n<NSCALARS; n++) total_error.s[n] += error.s[n];
623 #endif
624  }}

628  Nx2 = pM->Domain[0][0].Nx[1];
629  Nx3 = pM->Domain[0][0].Nx[2];
630 #else

632  Nx2 = je - js + 1;

634 #endif

636 
637 #ifdef MPI_PARALLEL

639  * grids. Begin by copying the error into the err[] array */
640

642  err[1] = total_error.M1;
643  err[2] = total_error.M2;

655 #endif

659  pM->Domain[0][0].Comm_Domain);
660

662

668  total_error.M3 = tot_err[3];

671  total_error.B2c = tot_err[5];
672  total_error.B3c = tot_err[6];
673 #endif /* MHD */

675  total_error.E = tot_err[7];
676 #endif /* ISOTHERMAL */
677 #if (NSCALARS > 0)
678  for (n=0; n<NSCALARS; n++) total_error.s[n] = tot_err.s[8+n];
679 #endif
680  }
681  else return; /* The child grids do not do any of the following code */
682 
683 #endif /* MPI_PARALLEL */
684 
685 /* Compute RMS error over all variables, and print out */
686 
687  rms_error = SQR(total_error.d) + SQR(total_error.M1) + SQR(total_error.M2)
688  + SQR(total_error.M3);
689 #ifdef MHD
690  rms_error += SQR(total_error.B1c) + SQR(total_error.B2c)
691  + SQR(total_error.B3c);
692 #endif /* MHD */

695 #endif /* ISOTHERMAL */

697  for (n=0; n<NSCALARS; n++) rms_error += SQR(total_error.s[n]);
698 #endif

702 /* For 1D, assume shock propagates along direction with MAX number of zones */
703 
704  min_zones = Nx1;
705  if (Nx2 > 1) min_zones = MAX(min_zones,Nx2);
706  if (Nx3 > 1) min_zones = MAX(min_zones,Nx3);
707  if (rms_error > 8.0/min_zones)
708  printf("WARNING: rms_error=%e exceeds estimate\n",rms_error);
