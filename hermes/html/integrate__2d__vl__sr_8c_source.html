838 #ifndef BAROTROPIC
839  pG->U[ks][j][i].E -= dtodx2*(x2Flux[j ][i].d*(phic - phil)
840  + x2Flux[j+1][i].d*(phir - phic));

860 #endif /* BAROTROPIC */

863 #endif /* MHD */

887 #endif /* BAROTROPIC */

908  */
909

911  for (i=is; i<=ie; i++) {
912  Wcheck = check_Prim(&(pG->U[ks][j][i]));

915  BadCell.i = i;

921  flag_cell = 1;
922  BadCell.i = i;
923  BadCell.j = j;
924  BadCell.k = ks;
925  negP++;
926  }
927  Vsq = SQR(Wcheck.V1) + SQR(Wcheck.V2) + SQR(Wcheck.V3);
928  if (Vsq > 1.0) {
929  flag_cell = 1;
930  BadCell.i = i;
931  BadCell.j = j;
932  BadCell.k = ks;
933  superl++;
934  }
935  if (flag_cell != 0) {
936  FixCell(pG, BadCell);

944  printf("[Step15a]: %i cells had v>1 at 1st order correction\n",superl);
945  }

947 /*--- Step 15b -----------------------------------------------------------------
948  * In SR the first-order flux correction can fail to fix an unphysical state.
949  * We must fix these cells in order to avoid NaN's at the next timestep,
950  * particuarly if v^2 > 1. We have 2 approaches; firstly, we use the entropy
951  * equation (which we have not applied a 1st order flux correction to) to
952  * calculate the pressure and the Lorentz factor of the gas. If this produces
953  * and unphysical state, then we floor the pressure and iterate on v^2 until
954  * v^2 < 1. Possibly could improved by averaging density and pressure from
955  * adjacent cells and then calculating pressure.
956  */
957 
958 #ifdef MHD
959  negd = 0;
960  negP = 0;
961  superl = 0;
962  entropy = 0;
963  final = 0;
964  fail = 0;
965  for (j=js; j<=je; j++) {
966  for (i=is; i<=ie; i++) {
967  flag_cell=0;
968  Wcheck = check_Prim(&(pG->U[ks][j][i]));
969  if (Wcheck.d < 0.0) {
970  flag_cell = 1;
971  negd++;
972  }
973  if (Wcheck.P < 0.0) {
974  flag_cell = 1;
975  negP++;
976  }
977  Vsq = SQR(Wcheck.V1) + SQR(Wcheck.V2) + SQR(Wcheck.V3);
978  if (Vsq > 1.0) {
979  flag_cell = 1;
980  superl++;
981  }
982 #ifdef USE_ENTROPY_FIX
983  if (flag_cell != 0) {
984  entropy++;
985  Wcheck = entropy_fix (&(pG->U[ks][j][i]),&(S[j][i]));
986  Ucheck = Prim_to_Cons(&Wcheck);
987  Wcheck = check_Prim(&Ucheck);
988  Vsq = SQR(Wcheck.V1) + SQR(Wcheck.V2) + SQR(Wcheck.V3);
989  if (Wcheck.d > 0.0 && Wcheck.P > 0.0 && Vsq < 1.0){
990  pG->U[ks][j][i].d = Ucheck.d;
991  pG->U[ks][j][i].M1 = Ucheck.M1;
992  pG->U[ks][j][i].M2 = Ucheck.M2;
993  pG->U[ks][j][i].M3 = Ucheck.M3;
994  pG->U[ks][j][i].E = Ucheck.E;
995  flag_cell = 0;
996  }
997  }
998 #endif
999  if (flag_cell != 0) {
1000  final++;
1001  Wcheck = fix_vsq (&(pG->U[ks][j][i]));
1002  Ucheck = Prim_to_Cons(&Wcheck);
1003  pG->U[ks][j][i].d = Ucheck.d;
1004  pG->U[ks][j][i].M1 = Ucheck.M1;
1005  pG->U[ks][j][i].M2 = Ucheck.M2;
1006  pG->U[ks][j][i].M3 = Ucheck.M3;
1007  pG->U[ks][j][i].E = Ucheck.E;
1008  Wcheck = check_Prim(&(pG->U[ks][j][i]));
1009  Vsq = SQR(Wcheck.V1) + SQR(Wcheck.V2) + SQR(Wcheck.V3);
1010  if (Wcheck.d < 0.0 || Wcheck.P < 0.0 || Vsq > 1.0){
1011  fail++;
1012  }
1013  }
1014  }
1015  }
1016 
1017  if (negd > 0 || negP > 0 || superl > 0) {
1018  printf("[Step15b]: %i cells had d<0; %i cells had P<0;\n",negd,negP);
1019  printf("[Step15b]: %i cells had v>1 after 1st order correction\n",superl);
1020  printf("[Step15b]: %i cells required an entropy fix\n",entropy);
1021  printf("[Step15b]: %i cells required a final fix\n",final);
1022  printf("[Step15b]: %i cells had an unphysical state\n",fail);
1023  }
1024 #endif /* MHD */
1025 #endif /* FIRST_ORDER_FLUX_CORRECTION */

1031  * store fluxes at boundaries of child grids.

1035 
1036 /* x1-boundaries of child Grids (interior to THIS Grid) */
