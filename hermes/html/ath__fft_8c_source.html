96 struct ath_3d_fft_plan *ath_3d_fft_create_plan(DomainS *pD, int gnx3, int gnx2,
97  int gnx1, int gks, int gke, int gjs, int gje,
98  int gis, int gie, ath_fft_data *data, int al,
99  ath_fft_direction dir)
100 {
101  int nbuf, tmp;
102  struct ath_3d_fft_plan *ath_plan;
103 
104  if ((dir != ATH_FFT_FORWARD) && (dir != ATH_FFT_BACKWARD)) {
105  ath_error("Invalid Athena FFT direction.\n");
106  }
107 
108  /* Allocate memory for the plan */
109  ath_plan = (struct ath_3d_fft_plan *)malloc(sizeof(struct ath_3d_fft_plan));
110  if (ath_plan == NULL) {
111  ath_error("Couldn't malloc for FFT plan.");
112  }
113  /* Set forward/backward FFT */
114  ath_plan->dir = dir;
115  /* Set element count (for easy malloc and memset) */
116  ath_plan->cnt = (gke-gks+1)*(gje-gjs+1)*(gie-gis+1);
117  ath_plan->gcnt = gnx3*gnx2*gnx1;
118 
119  tmp = (al==0 ? 1 : 0);
120  if (data != NULL) tmp = 0;
121 
122  /* If data == NULL, then allocate something (temporarily if tmp=1) */
123  if (data == NULL)
124  data = (ath_fft_data *)ath_3d_fft_malloc(ath_plan);
125  if (data == NULL)
126  ath_error("Couln't malloc for FFT plan data.");
127 
128  /* Create the plan */
129 #ifdef FFT_BLOCK_DECOMP
130  /* Block decomp library plans don't care if forward or backward */
131  ath_plan->plan = fft_3d_create_plan(pD->Comm_Domain, gnx3, gnx2, gnx1,
132  gks, gke, gjs, gje, gis, gie,
133  gks, gke, gjs, gje, gis, gie,
134  0, 0, &nbuf);
135 #else /* FFT_BLOCK_DECOMP */
136  if (dir == ATH_FFT_FORWARD) {
137  ath_plan->plan = fftw_plan_dft_3d(gnx1, gnx2, gnx3, data, data,
138  FFTW_FORWARD, FFTW_MEASURE);
139  } else {
140  ath_plan->plan = fftw_plan_dft_3d(gnx1, gnx2, gnx3, data, data,
141  FFTW_BACKWARD, FFTW_MEASURE);
142  }
143 #endif /* FFT_BLOCK_DECOMP */
144 
145  if (tmp) ath_3d_fft_free(data);
146 
147  return ath_plan;
148 }
