186  Real arg,arm_amp,Phi_bg,Phi_sp;
187  arg = x1*x1 + 1.;
188  Phi_bg = SQR(vcir)*0.5*(1./arg + log(arg));

192  if(x1 < taper_i) arm_amp =ephi*exp(-5.*SQR(x1-taper_i));
