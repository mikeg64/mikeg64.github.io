46  js = pGrid->js; je = pGrid->je;

50

53  drat = par_getd("problem","drat");

57 #endif

59 /* iprob=1. Two uniform streams moving at +/- vflow, random perturbations */
60 
61  if (iprob == 1) {
62  for (k=ks; k<=ke; k++) {
63  for (j=js; j<=je; j++) {
64  for (i=is; i<=ie; i++) {
65  cc_pos(pGrid,i,j,k,&x1,&x2,&x3);
66  pGrid->U[k][j][i].d = 1.0;
67  pGrid->U[k][j][i].M1 = vflow + amp*(ran2(&iseed) - 0.5);
68  pGrid->U[k][j][i].M2 = amp*(ran2(&iseed) - 0.5);
69  pGrid->U[k][j][i].M3 = 0.0;
70  if (fabs(x2) < 0.25) {
71  pGrid->U[k][j][i].d = drat;
72  pGrid->U[k][j][i].M1 = -drat*(vflow + amp*(ran2(&iseed) - 0.5));
73  pGrid->U[k][j][i].M2 = drat*amp*(ran2(&iseed) - 0.5);
74  }
75 /* Pressure scaled to give a sound speed of 1 with gamma=1.4 */
76 #ifndef BAROTROPIC
77  pGrid->U[k][j][i].E = 2.5/Gamma_1
78  + 0.5*(SQR(pGrid->U[k][j][i].M1) + SQR(pGrid->U[k][j][i].M2)
79  + SQR(pGrid->U[k][j][i].M3))/pGrid->U[k][j][i].d;
80 #endif /* BAROTROPIC */
81 #ifdef MHD
82  pGrid->B1i[k][j][i] = b0;
83  pGrid->U[k][j][i].B1c = b0;
84 #ifndef BAROTROPIC
85  pGrid->U[k][j][i].E += 0.5*b0*b0;
86 #endif /* BAROTROPIC */
87 #endif /* MHD */
88  }
89 #ifdef MHD

91 #endif
92  }
93  }
94  }

96 /* iprob=2. Test suggested by E. Zweibel, based on Ryu & Jones.
97  * Two uniform density flows with single mode perturbation
98  */
99 
100  if (iprob == 2) {
101  a = 0.05;
102  sigma = 0.2;
103  for (k=ks; k<=ke; k++) {
104  for (j=js; j<=je; j++) {
105  for (i=is; i<=ie; i++) {
106  cc_pos(pGrid,i,j,k,&x1,&x2,&x3);
107  pGrid->U[k][j][i].d = 1.0;
108  pGrid->U[k][j][i].M1 = vflow*tanh(x2/a);
109  pGrid->U[k][j][i].M2 = amp*sin(2.0*PI*x1)*exp(-(x2*x2)/(sigma*sigma));
110  pGrid->U[k][j][i].M3 = 0.0;
111 #ifndef BAROTROPIC
112  pGrid->U[k][j][i].E = 1.0/Gamma_1
113  + 0.5*(SQR(pGrid->U[k][j][i].M1) + SQR(pGrid->U[k][j][i].M2)
114  + SQR(pGrid->U[k][j][i].M3))/pGrid->U[k][j][i].d;
115 #endif /* BAROTROPIC */
116 #ifdef MHD
117  pGrid->B1i[k][j][i] = b0;
118  pGrid->U[k][j][i].B1c = b0;
119 #ifndef BAROTROPIC
120  pGrid->U[k][j][i].E += 0.5*b0*b0;
121 #endif /* BAROTROPIC */
122 #endif /* MHD */
123 /* Use passive scalar to keep track of the fluids, since densities are same */
124 #if (NSCALARS > 0)
125  pGrid->U[k][j][i].s[0] = 0.0;
126  if (x2 > 0) pGrid->U[k][j][i].s[0] = 1.0;
127 #endif
128  }
129 #ifdef MHD
130  pGrid->B1i[k][j][ie+1] = b0;
131 #endif
132  }
133  }
134  }

253 double ran2(long int *idum){
254  int j;
255  long int k;
256  static long int idum2=123456789;
257  static long int iy=0;
258  static long int iv[NTAB];
259  double temp;
260 
261  if (*idum <= 0) { /* Initialize */
262  if (-(*idum) < 1) *idum=1; /* Be sure to prevent idum = 0 */
263  else *idum = -(*idum);
264  idum2=(*idum);
265  for (j=NTAB+7;j>=0;j--) { /* Load the shuffle table (after 8 warm-ups) */
266  k=(*idum)/IQ1;
267  *idum=IA1*(*idum-k*IQ1)-k*IR1;
268  if (*idum < 0) *idum += IM1;
269  if (j < NTAB) iv[j] = *idum;
270  }
271  iy=iv[0];
272  }
273  k=(*idum)/IQ1; /* Start here when not initializing */
274  *idum=IA1*(*idum-k*IQ1)-k*IR1; /* Compute idum=(IA1*idum) % IM1 without */
275  if (*idum < 0) *idum += IM1; /* overflows by Schrage's method */
276  k=idum2/IQ2;
277  idum2=IA2*(idum2-k*IQ2)-k*IR2; /* Compute idum2=(IA2*idum) % IM2 likewise */
278  if (idum2 < 0) idum2 += IM2;
279  j=(int)(iy/NDIV); /* Will be in the range 0...NTAB-1 */
280  iy=iv[j]-idum2; /* Here idum is shuffled, idum and idum2 */
281  iv[j] = *idum; /* are combined to generate output */
282  if (iy < 1) iy += IMM1;
283  if ((temp=AM*iy) > RNMX) return RNMX; /* No endpoint values */
284  else return temp;
285 }
