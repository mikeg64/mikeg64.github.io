355 static Real3Vect ParticleTroj(Real t)
356 {
357  Real3Vect pos;
358  Real L1,L2,L3;
359  Real factor=tstop0[0]/(1.0+mratio)*(1.0-exp(-(1.0+mratio)*t/tstop0[0]));
360 
361  pos.x1 = x0[0] + wx*factor;
362  pos.x2 = x0[1] + wy*factor;
363  pos.x3 = x0[2];
364 
365  L1 = x1max-x1min; L2 = x2max-x2min;
366 
367  pos.x1 = pos.x1-floor((pos.x1-x1min)/L1)*L1;
368  pos.x2 = pos.x2-floor((pos.x2-x2min)/L2)*L2;
369 
370  return pos;
371 }
