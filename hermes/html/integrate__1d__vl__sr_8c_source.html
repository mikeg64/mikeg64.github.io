166 #endif /* MHD */

616 static void FixCell(GridS *pG, Int3Vect indx)
617 {
618  int ks=pG->ks,js=pG->js;
619  Real dtodx1=pG->dt/pG->dx1;
620  Cons1DS x1FD_i, x1FD_ip1, x2FD_j, x2FD_jp1;
621 
622  /* Compute difference of predictor and corrector fluxes at cell faces */
623 
624  x1FD_i.d = x1Flux[indx.i ].d - x1FluxP[indx.i ].d;
625  x1FD_ip1.d = x1Flux[indx.i+1].d - x1FluxP[indx.i+1].d;
626 
627  x1FD_i.Mx = x1Flux[indx.i ].Mx - x1FluxP[indx.i ].Mx;
628  x1FD_ip1.Mx = x1Flux[indx.i+1].Mx - x1FluxP[indx.i+1].Mx;
629 
630  x1FD_i.My = x1Flux[indx.i ].My - x1FluxP[indx.i ].My;
631  x1FD_ip1.My = x1Flux[indx.i+1].My - x1FluxP[indx.i+1].My;
632 
633  x1FD_i.Mz = x1Flux[indx.i ].Mz - x1FluxP[indx.i ].Mz;
634  x1FD_ip1.Mz = x1Flux[indx.i+1].Mz - x1FluxP[indx.i+1].Mz;
635 
636 #ifdef MHD
637  x1FD_i.By = x1Flux[indx.i ].By - x1FluxP[indx.i ].By;
638  x1FD_ip1.By = x1Flux[indx.i+1].By - x1FluxP[indx.i+1].By;
639 
640  x1FD_i.Bz = x1Flux[indx.i ].Bz - x1FluxP[indx.i ].Bz;
641  x1FD_ip1.Bz = x1Flux[indx.i+1].Bz - x1FluxP[indx.i+1].Bz;
642 #endif
643 
644 #ifndef BAROTROPIC
645  x1FD_i.E = x1Flux[indx.i ].E - x1FluxP[indx.i ].E;
646  x1FD_ip1.E = x1Flux[indx.i+1].E - x1FluxP[indx.i+1].E;
647 #endif /* BAROTROPIC */
648 
649  /* Use flux differences to correct bad cell */
650  pG->U[ks][js][indx.i].d += dtodx1*(x1FD_ip1.d - x1FD_i.d );
651  pG->U[ks][js][indx.i].M1 += dtodx1*(x1FD_ip1.Mx - x1FD_i.Mx);
652  pG->U[ks][js][indx.i].M2 += dtodx1*(x1FD_ip1.My - x1FD_i.My);
653  pG->U[ks][js][indx.i].M3 += dtodx1*(x1FD_ip1.Mz - x1FD_i.Mz);
654 #ifdef MHD
655  pG->U[ks][js][indx.i].B2c += dtodx1*(x1FD_ip1.By - x1FD_i.By);
656  pG->U[ks][js][indx.i].B3c += dtodx1*(x1FD_ip1.Bz - x1FD_i.Bz);
657  /* For consistency, set B2i and B3i to cell-centered values. */
658  pG->B2i[ks][js][indx.i] = pG->U[ks][js][indx.i].B2c;
659  pG->B3i[ks][js][indx.i] = pG->U[ks][js][indx.i].B3c;
660 #endif
661 #ifndef BAROTROPIC
662  pG->U[ks][js][indx.i].E += dtodx1*(x1FD_ip1.E - x1FD_i.E );
663 #endif /* BAROTROPIC */
664 
665 
666  /* Use flux differences to correct bad cell neighbors at i-1 and i+1 */
667  if (indx.i > pG->is) {
668  pG->U[ks][js][indx.i-1].d += dtodx1*(x1FD_i.d );
669  pG->U[ks][js][indx.i-1].M1 += dtodx1*(x1FD_i.Mx);
670  pG->U[ks][js][indx.i-1].M2 += dtodx1*(x1FD_i.My);
671  pG->U[ks][js][indx.i-1].M3 += dtodx1*(x1FD_i.Mz);
672 #ifdef MHD
673  pG->U[ks][js][indx.i-1].B2c += dtodx1*(x1FD_i.By);
674  pG->U[ks][js][indx.i-1].B3c += dtodx1*(x1FD_i.Bz);
675  /* For consistency, set B2i and B3i to cell-centered values. */
676  pG->B2i[ks][js][indx.i-1] = pG->U[ks][js][indx.i-1].B2c;
677  pG->B3i[ks][js][indx.i-1] = pG->U[ks][js][indx.i-1].B3c;
678 #endif
679 #ifndef BAROTROPIC
680  pG->U[ks][js][indx.i-1].E += dtodx1*(x1FD_i.E );
681 #endif /* BAROTROPIC */
682  }
683 
684  if (indx.i < pG->ie) {
685  pG->U[ks][js][indx.i+1].d -= dtodx1*(x1FD_ip1.d );
686  pG->U[ks][js][indx.i+1].M1 -= dtodx1*(x1FD_ip1.Mx);
687  pG->U[ks][js][indx.i+1].M2 -= dtodx1*(x1FD_ip1.My);
688  pG->U[ks][js][indx.i+1].M3 -= dtodx1*(x1FD_ip1.Mz);
689 #ifdef MHD
690  pG->U[ks][js][indx.i+1].B2c -= dtodx1*(x1FD_ip1.By);
691  pG->U[ks][js][indx.i+1].B3c -= dtodx1*(x1FD_ip1.Bz);
692  /* For consistency, set B2i and B3i to cell-centered values. */
693  pG->B2i[ks][js][indx.i+1] = pG->U[ks][js][indx.i+1].B2c;
694  pG->B3i[ks][js][indx.i+1] = pG->U[ks][js][indx.i+1].B3c;
695 #endif /* MHD */
696 #ifndef BAROTROPIC
697  pG->U[ks][js][indx.i+1].E -= dtodx1*(x1FD_ip1.E );
698 #endif /* BAROTROPIC */
699  }
