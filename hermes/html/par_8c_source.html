160  char *block_name, *cp;
161  Block *bp = NULL;
162 
163  if (now_open) ath_error("Parameter file %s still open\n",now_filename);
164  if (now_filename) free(now_filename);

181  continue;
182  }

184  }

669 static void add_par_line(Block *bp, char *line)
670 {
671  char *cp;
672  char *name, *equal=NULL, *value=NULL, *hash=NULL, *comment=NULL, *nul;
673 
674  if(bp == NULL)
675  ath_error("[add_par_line]: (no block name) while parsing line \n%s\n",line);
676 
677  name = skipwhite(line); /* name */
678 
679  for(cp = name; *cp != '\0'; cp++){/* Find the first '=' and '#' */
680  if(*cp == '='){
681  if(equal == NULL){
682  equal = cp; /* store the equals sign location */
683  value = skipwhite(cp + 1); /* value */
684  }
685  }
686  if(*cp == '#'){
687  hash = cp; /* store the hash sign location */
688  comment = skipwhite(cp + 1); /* comment */
689  break;
690  }
691  }
692 
693  while(*cp != '\0') cp++; /* Find the NUL terminator */
694  nul = cp;
695 
696  if(equal == NULL)
697  ath_error("No '=' found in line \"%s\"\n",line);
698 
699  str_term(equal); /* Terminate the name string */
700 
701  if(hash == NULL){
702  str_term(nul); /* Terminate the value string */
703  }
704  else{
705  str_term(hash); /* Terminate the value string */
706 
707  if(*comment == '\0')
708  comment = NULL; /* Comment field is empty */
709  else
710  str_term(nul); /* Terminate the comment string */
711  }
712 
713  add_par(bp,name,value,comment);
714 }
