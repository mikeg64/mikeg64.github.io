677  time1 = tbuf.tms_utime + tbuf.tms_stime;
678  cpu_time = (time1 > time0 ? (double)(time1 - time0) : 1.0)/
679  (double)clk_tck;
680  } else {
681  time1 = clock();
682  cpu_time = (time1 > time0 ? (double)(time1 - time0) : 1.0)/
683  (double)CLOCKS_PER_SEC;
684  }
685 
686 /* Calculate and print the zone-cycles/cpu-second on this processor */
687 
688  zones = 0;
689  for (nl=0; nl<(Mesh.NLevels); nl++){
690  for (nd=0; nd<(Mesh.DomainsPerLevel[nl]); nd++){
691  if (Mesh.Domain[nl][nd].Grid != NULL) {
692  zones += (Mesh.Domain[nl][nd].Grid->Nx[0])*
693  (Mesh.Domain[nl][nd].Grid->Nx[1])*
694  (Mesh.Domain[nl][nd].Grid->Nx[2]);
695  }
696  }}
697  zcs = (double)zones*(double)((Mesh.nstep) - nstep_start)/cpu_time;
698 
699  ath_pout(0," tlim= %e nlim= %i\n",tlim,nlim);
700  ath_pout(0," time= %e cycle= %i\n",Mesh.time,Mesh.nstep);
701  ath_pout(0,"\nzone-cycles/cpu-second = %e\n",zcs);
702 
703 /* Calculate and print the zone-cycles/wall-second on this processor */
704 
705  cpu_time = (double)(tve.tv_sec - tvs.tv_sec) +
706  1.0e-6*(double)(tve.tv_usec - tvs.tv_usec);
707  zcs = (double)zones*(double)((Mesh.nstep) - nstep_start)/cpu_time;
708  ath_pout(0,"\nelapsed wall time = %e sec.\n",cpu_time);
709  ath_pout(0,"\nzone-cycles/wall-second = %e\n",zcs);
710 
711 /* Calculate and print total zone-cycles/wall-second on all processors */
712 #ifdef MPI_PARALLEL
713  zones = 0;
714  for (nl=0; nl<(Mesh.NLevels); nl++){
715  for (nd=0; nd<(Mesh.DomainsPerLevel[nl]); nd++){
716  zones += (Mesh.Domain[nl][nd].Nx[0])*
717  (Mesh.Domain[nl][nd].Nx[1])*
718  (Mesh.Domain[nl][nd].Nx[2]);
719  }}
720  zcs = (double)zones*(double)(Mesh.nstep - nstep_start)/cpu_time;
721  ath_pout(0,"\ntotal zone-cycles/wall-second = %e\n",zcs);
722 #endif /* MPI_PARALLEL */
723 
724 /* complete any final User work */
725 
726  Userwork_after_loop(&Mesh);
727 
728 /* Final output everything (last argument of data_output = 1) */
729 
730  data_output(&Mesh, 1);
731 
732 /* Free all memory */
733 
734  lr_states_destruct();
735  integrate_destruct();
736  data_output_destruct();
737 #ifdef PARTICLES
738  particle_destruct(&Mesh);
739  bvals_particle_destruct(&Mesh);
740 #endif
741 #if defined(SHEARING_BOX) || (defined(FARGO) && defined(CYLINDRICAL))
742  bvals_shear_destruct();
743 #endif
744 #if defined(RESISTIVITY) || defined(VISCOSITY) || defined(THERMAL_CONDUCTION)
745  integrate_diff_destruct();
746 #endif
747  par_close();
748

750  MPI_Finalize();
751 #endif /* MPI_PARALLEL */
752 
753  if(time(&stop)>0) /* current calendar time (UTC) is available */
754  ath_pout(0,"\nSimulation terminated on %s",ctime(&stop));
755 
756  ath_log_close(); /* close the simulation log files */
757 
758  return EXIT_SUCCESS;

767 void change_rundir(const char *name)
768 {
769 #ifdef MPI_PARALLEL
770 
771  int err=0;
772  int rerr, gerr, my_id, status;
773  char mydir[80];
774 
775  status = MPI_Comm_rank(MPI_COMM_WORLD, &my_id);
776  if(status != MPI_SUCCESS)
777  ath_error("[change_rundir]: MPI_Comm_rank error = %d\n",status);
778 
779  if(name != NULL && *name != '\0'){
780 
781  if(my_id == 0)
782  mkdir(name, 0775); /* May return an error, e.g. the directory exists */
783 
784  MPI_Barrier(MPI_COMM_WORLD); /* Wait for rank 0 to mkdir() */
785 
786  baton_start(MAX_FILE_OP, ch_rundir0_tag);
787 
788  if(chdir(name)){
789  ath_perr(-1,"[change_rundir]: Cannot change directory to \"%s\"\n",name);
790  err = 1;
791  }
792 
793  baton_stop(MAX_FILE_OP, ch_rundir0_tag);
794 
795  /* Did anyone fail to make and change to the run directory? */
796  rerr = MPI_Allreduce(&err, &gerr, 1, MPI_INT, MPI_MAX, MPI_COMM_WORLD);
797  if(rerr) ath_perr(-1,"[change_rundir]: MPI_Allreduce error = %d\n",rerr);
798 
799  if(rerr || gerr){
800  MPI_Abort(MPI_COMM_WORLD, 1);
801  exit(EXIT_FAILURE);
802  }
803  }
804 
805 /* Next, change to the local run directory */
806 
807  sprintf(mydir, "id%d", my_id);
808 
809  baton_start(MAX_FILE_OP, ch_rundir1_tag);
810 
811  mkdir(mydir, 0775); /* May return an error, e.g. the directory exists */
812  if(chdir(mydir)){
813  ath_perr(-1,"[change_rundir]: Cannot change directory to \"%s\"\n",mydir);
814  err = 1;
815  }
816 
817  baton_stop(MAX_FILE_OP, ch_rundir1_tag);
818 
819  /* Did anyone fail to make and change to the local run directory? */
820  rerr = MPI_Allreduce(&err, &gerr, 1, MPI_INT, MPI_MAX, MPI_COMM_WORLD);
821  if(rerr) ath_perr(-1,"[change_rundir]: MPI_Allreduce error = %d\n",rerr);
822 
823  if(rerr || gerr){
824  MPI_Abort(MPI_COMM_WORLD, 1);
825  exit(EXIT_FAILURE);
826  }
827 
828 #else /* Serial job */
829 
830  if(name == NULL || *name == '\0') return;
831 
832  mkdir(name, 0775); /* May return an error, e.g. the directory exists */
833  if(chdir(name))
834  ath_error("[change_rundir]: Cannot change directory to \"%s\"\n",name);
