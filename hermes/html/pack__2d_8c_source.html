43 void pack_2d(double *data, double *buf, struct pack_plan_2d *plan)
44 
45 {
46  register int in,out,fast,slow;
47  register int nfast,nslow,nstride;
48 
49  nfast = plan->nfast;
50  nslow = plan->nslow;
51  nstride = plan->nstride;
52 
53  in = 0;
54  for (slow = 0; slow < nslow; slow++) {
55  out = slow*nstride;
