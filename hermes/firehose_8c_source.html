135 #ifdef MHD

246 double ran2(long int *idum)
247 {
248  int j;
249  long int k;
250  static long int idum2=123456789;
251  static long int iy=0;
252  static long int iv[NTAB];
253  double temp;
254 
255  if (*idum <= 0) { /* Initialize */
256  if (-(*idum) < 1) *idum=1; /* Be sure to prevent idum = 0 */
257  else *idum = -(*idum);
258  idum2=(*idum);
259  for (j=NTAB+7;j>=0;j--) { /* Load the shuffle table (after 8 warm-ups) */
260  k=(*idum)/IQ1;
261  *idum=IA1*(*idum-k*IQ1)-k*IR1;
262  if (*idum < 0) *idum += IM1;
263  if (j < NTAB) iv[j] = *idum;
264  }
265  iy=iv[0];
266  }
267  k=(*idum)/IQ1; /* Start here when not initializing */
268  *idum=IA1*(*idum-k*IQ1)-k*IR1; /* Compute idum=(IA1*idum) % IM1 without */
269  if (*idum < 0) *idum += IM1; /* overflows by Schrage's method */
270  k=idum2/IQ2;
271  idum2=IA2*(idum2-k*IQ2)-k*IR2; /* Compute idum2=(IA2*idum) % IM2 likewise */
272  if (idum2 < 0) idum2 += IM2;
273  j=(int)(iy/NDIV); /* Will be in the range 0...NTAB-1 */
274  iy=iv[j]-idum2; /* Here idum is shuffled, idum and idum2 */
275  iv[j] = *idum; /* are combined to generate output */
276  if (iy < 1) iy += IMM1;
277  if ((temp=AM*iy) > RNMX) return RNMX; /* No endpoint values */
278  else return temp;
279 }
