452 void ROOTJ(int N, int NK, double *JZERO, int *IER) {
453 /*----------------------------------------------------------------------
454 ! CALCULATE THE FIRST NK ZEROES OF BESSEL FUNCTION J(N,X)
455 !
456 ! INPUTS:
457 ! N ORDER OF FUNCTION J (INTEGER >= 0) I*4
458 ! NK NUMBER OF FIRST ZEROES (INTEGER > 0) I*4
459 ! OUTPUTS:
460 ! JZERO(NK) TABLE OF FIRST ZEROES (ABCISSAS) R*8
461 ! IER(NK) TABLE OF ERROR CODES (MUST BE ZEROES) I*4
462 !
463 ! REFERENCE :
464 ! ABRAMOWITZ M. & STEGUN IRENE A.
465 ! HANDBOOK OF MATHEMATICAL FUNCTIONS
466 ! -------------------------------------------------------------------- */
467  double ZEROJ,B0,B1,B2,B3,B5,B7,T0,T1,T3,T5,T7,FN,FK;
468  double C1,C2,C3,C4,C5,F1,F2,F3,TOL,ERRJ;
469  int IERROR,K,NITMX;
470 
471  TOL=1E-8; NITMX=10;
472  C1=1.8557571; C2=1.033150; C3=0.00397; C4=0.0908; C5=0.043;
473  FN = 1.0*N;
474 
475 // FIRST ZERO
476  if (N==0) {
477  ZEROJ = C1+C2-C3-C4+C5;
478  SECANT(N,NITMX,TOL,&ZEROJ,&IERROR);
479  IER[1]=IERROR;
480  JZERO[1]=ZEROJ;
481  }
482  else {
483  F1 = pow(FN,(1.0/3.0));
484  F2 = F1*F1*FN;
485  F3 = F1*FN*FN;
486  ZEROJ = FN+C1*F1+(C2/F1)-(C3/FN)-(C4/F2)+(C5/F3);
487  SECANT(N,NITMX,TOL,&ZEROJ,&IERROR);
488  IER[1]=IERROR;
489  JZERO[1]=ZEROJ;
490  }
491 
492  T0 = 4.0*FN*FN;
493  T1 = T0-1.0;
494  T3 = 4.0*T1*(7.0*T0-31.0);
495  T5 = 32.0*T1*((83.0*T0-982.0)*T0+3779.0);
496  T7 = 64.0*T1*(((6949.0*T0-153855.0)*T0+1585743.0)*T0-6277237.0);
497 
498 // OTHER ZEROES
499  for (K = 2; K <= NK; K++) {
500  JZERO[K] = 0.0;
501  FK = 1.0*K;
502 
503 // MAC MAHON'S SERIES FOR K>>N
504 
505  B0 = (FK+0.5*FN-0.25)*PI;
506  B1 = 8.0*B0;
507  B2 = B1*B1;
508  B3 = 3.0*B1*B2;
509  B5 = 5.0*B3*B2;
510  B7 = 7.0*B5*B2;
511 
512  ZEROJ = B0-(T1/B1)-(T3/B3)-(T5/B5)-(T7/B7);
513 
514  ERRJ=fabs(jn(N,ZEROJ));
515 
516 // IMPROVE SOLUTION USING PROCEDURE SECANT
517 
518  if (ERRJ > TOL) SECANT(N,NITMX,TOL,&ZEROJ,&IERROR);
519  JZERO[K]=ZEROJ;
520  IER[K]=IERROR;
521  }
522 }
