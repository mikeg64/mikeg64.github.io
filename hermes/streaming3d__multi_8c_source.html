503 void MultiNSH(int n, Real *tstop, Real *mratio, Real etavk,
504  Real *uxNSH, Real *uyNSH, Real *wxNSH, Real *wyNSH)
505 {
506  int i,j;
507  Real *Lambda1,**Lam1GamP1, **A, **B, **Tmp;
508 
509  Lambda1 = (Real*)calloc_1d_array(n, sizeof(Real)); /* Lambda^{-1} */
510  Lam1GamP1=(Real**)calloc_2d_array(n, n, sizeof(Real)); /* Lambda1*(1+Gamma) */
511  A = (Real**)calloc_2d_array(n, n, sizeof(Real));
512  B = (Real**)calloc_2d_array(n, n, sizeof(Real));
513  Tmp = (Real**)calloc_2d_array(n, n, sizeof(Real));
514 
515  /* definitions */
516  for (i=0; i<n; i++){
517  for (j=0; j<n; j++)
518  Lam1GamP1[i][j] = mratio[j];
519  Lam1GamP1[i][i] += 1.0;
520  Lambda1[i] = 1.0/(tstop[i]+1.0e-16);
521  for (j=0; j<n; j++)
522  Lam1GamP1[i][j] *= Lambda1[i];
523  }
524 
525  /* Calculate A and B */
526  MatrixMult(Lam1GamP1, Lam1GamP1, n,n,n, Tmp);
527  for (i=0; i<n; i++) Tmp[i][i] += 1.0;
528  InverseMatrix(Tmp, n, B);
529  for (i=0; i<n; i++)
530  for (j=0; j<n; j++)
531  B[i][j] *= Lambda1[j];
532  MatrixMult(Lam1GamP1, B, n,n,n, A);
533 
534  /* Obtain NSH velocities */
535  *uxNSH = 0.0; *uyNSH = 0.0;
536  for (i=0; i<n; i++){
537  wxNSH[i] = 0.0;
538  wyNSH[i] = 0.0;
539  for (j=0; j<n; j++){
540  wxNSH[i] -= B[i][j];
541  wyNSH[i] -= A[i][j];
542  }
543  wxNSH[i] *= 2.0*etavk;
544  wyNSH[i] *= etavk;
545  *uxNSH -= mratio[i]*wxNSH[i];
546  *uyNSH -= mratio[i]*wyNSH[i];
547  wyNSH[i] += etavk;
548  }
549 
550  free(Lambda1);
551  free_2d_array(A); free_2d_array(B);
552  free_2d_array(Lam1GamP1); free_2d_array(Tmp);
553 
554  return;
555 }

588 double ran2(long int *idum)
589 {
590  int j;
591  long int k;
592  static long int idum2=123456789;
593  static long int iy=0;
594  static long int iv[NTAB];
595  double temp;
596 
597  if (*idum <= 0) { /* Initialize */
598  if (-(*idum) < 1) *idum=1; /* Be sure to prevent idum = 0 */
599  else *idum = -(*idum);
600  idum2=(*idum);
601  for (j=NTAB+7;j>=0;j--) { /* Load the shuffle table (after 8 warm-ups) */
602  k=(*idum)/IQ1;
603  *idum=IA1*(*idum-k*IQ1)-k*IR1;
604  if (*idum < 0) *idum += IM1;
605  if (j < NTAB) iv[j] = *idum;
606  }
607  iy=iv[0];
608  }
609  k=(*idum)/IQ1; /* Start here when not initializing */
610  *idum=IA1*(*idum-k*IQ1)-k*IR1; /* Compute idum=(IA1*idum) % IM1 without */
611  if (*idum < 0) *idum += IM1; /* overflows by Schrage's method */
612  k=idum2/IQ2;
613  idum2=IA2*(idum2-k*IQ2)-k*IR2; /* Compute idum2=(IA2*idum) % IM2 likewise */
614  if (idum2 < 0) idum2 += IM2;
615  j=(int)(iy/NDIV); /* Will be in the range 0...NTAB-1 */
616  iy=iv[j]-idum2; /* Here idum is shuffled, idum and idum2 */
617  iv[j] = *idum; /* are combined to generate output */
618  if (iy < 1) iy += IMM1;
619  if ((temp=AM*iy) > RNMX) return RNMX; /* No endpoint values */
620  else return temp;
621 }
