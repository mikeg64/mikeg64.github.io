68 void Integrate_Particles(DomainS *pD)
69 {
70  GrainS *curG, *curP, mygr; /* pointer of the current working position */
71  long p; /* particle index */
72  Real dv1, dv2, dv3, ts, t1; /* amount of velocity update, stopping time */
73  Real3Vect cell1; /* one over dx1, dx2, dx3 */
74 
75  GridS *pG = pD->Grid; /* set ptr to Grid */
76 
77 /* Initialization */
78 #ifdef FEEDBACK
79  feedback_clear(pG); /* clean the feedback array */
80 #endif /* FEEDBACK */
81 
82  curP = &(mygr); /* temperory particle */
83 
84  /* cell1 is a shortcut expressions as well as dimension indicator */
85  if (pG->Nx[0] > 1) cell1.x1 = 1.0/pG->dx1; else cell1.x1 = 0.0;
86  if (pG->Nx[1] > 1) cell1.x2 = 1.0/pG->dx2; else cell1.x2 = 0.0;
87  if (pG->Nx[2] > 1) cell1.x3 = 1.0/pG->dx3; else cell1.x3 = 0.0;
88 
89  /* delete all ghost particles */
90  Delete_Ghost(pG);
91 
92  p = 0;
93  while (p<pG->nparticle)
94  {/* loop over all particles */
95  curG = &(pG->particle[p]);
96 
97 /* Step 1: Calculate velocity update */
98  switch(grproperty[curG->property].integrator)
99  {
100  case 1: /* 2nd order explicit integrator */
101  int_par_exp(pG, curG, cell1, &dv1, &dv2, &dv3, &ts);
102  break;
103 
104  case 2: /* 2nd order semi-implicit integrator */
105  int_par_semimp(pG, curG, cell1, &dv1, &dv2, &dv3, &ts);
106  break;
107 
108  case 3: /* 2nd order fully implicit integrator */
109  int_par_fulimp(pG, curG, cell1, &dv1, &dv2, &dv3, &ts);
110  break;
111 
112  default:
113  ath_error("[integrate_particle]: unknown integrator type!");
114  }
115 
116 /* Step 2: particle update to curP */
117 
118  /* velocity update */
119  curP->v1 = curG->v1 + dv1;
120  curP->v2 = curG->v2 + dv2;
121  curP->v3 = curG->v3 + dv3;
122 
123  /* position update */
124  if (pG->Nx[0] > 1)
125  curP->x1 = curG->x1 + 0.5*pG->dt*(curG->v1 + curP->v1);
126  else /* do not move if this dimension collapses */
127  curP->x1 = curG->x1;
128 
129  if (pG->Nx[1] > 1)
130  curP->x2 = curG->x2 + 0.5*pG->dt*(curG->v2 + curP->v2);
131  else /* do not move if this dimension collapses */
132  curP->x2 = curG->x2;
133 
134  if (pG->Nx[2] > 1)
135  curP->x3 = curG->x3 + 0.5*pG->dt*(curG->v3 + curP->v3);
136  else /* do not move if this dimension collapses */
137  curP->x3 = curG->x3;
138 
139 #ifdef FARGO
140  /* shift = -qshear * Omega_0 * x * dt */
141  pG->parsub[p].shift = -0.5*qshear*Omega_0*(curG->x1+curP->x1)*pG->dt;
142 #endif
143 
144 /* Step 3: calculate feedback force to the gas */
145 #ifdef FEEDBACK
146  feedback_corrector(pG, curG, curP, cell1, dv1, dv2, dv3, ts);
147 #endif /* FEEDBACK */
148 
149 /* Step 4: Final update of the particle */
150  /* update particle status (crossing boundary or not) */
151  JudgeCrossing(pG, curP->x1, curP->x2, curP->x3, curG);
152 
153  /* update the particle */
154  curG->x1 = curP->x1;
155  curG->x2 = curP->x2;
156  curG->x3 = curP->x3;
157  curG->v1 = curP->v1;
158  curG->v2 = curP->v2;
159  curG->v3 = curP->v3;
160  p++;
161 
162  } /* end of the for loop */
163 
164  /* output the status */
165  ath_pout(0, "In processor %d, there are %ld particles.\n",
166  myID_Comm_world, pG->nparticle);
167 
168  return;

181 void int_par_fulimp(GridS *pG, GrainS *curG, Real3Vect cell1,
182  Real *dv1, Real *dv2, Real *dv3, Real *ts)
183 {
184  Real x1n, x2n, x3n; /* first order new position at half a time step */
185  Real3Vect fd, fr; /* drag force and other forces */
186  Real3Vect fc, fp, ft; /* force at current & predicted position, total force */
187  Real ts11, ts12; /* 1/stopping time */
188  Real b0,A,B,C,D,Det1; /* matrix elements and determinant */
189 #ifdef SHEARING_BOX
190  Real oh, oh2; /* Omega_0*dt and its square */
191 #endif
192 
193 /* step 1: predict of the particle position after one time step */
194  if (pG->Nx[0] > 1) x1n = curG->x1+curG->v1*pG->dt;
195  else x1n = curG->x1;
196  if (pG->Nx[1] > 1) x2n = curG->x2+curG->v2*pG->dt;
197  else x2n = curG->x2;
198  if (pG->Nx[2] > 1) x3n = curG->x3+curG->v3*pG->dt;
199  else x3n = curG->x3;
200 
201 #ifdef SHEARING_BOX
202 #ifndef FARGO
203  /* advection part */
204  if (ShBoxCoord == xy) x2n -= 0.5*qshear*curG->v1*SQR(pG->dt);
205 #endif
206 #endif
207 
208 /* step 2: calculate the force at current position */
209  fd = Get_Drag(pG, curG->property, curG->x1, curG->x2, curG->x3,
210  curG->v1, curG->v2, curG->v3, cell1, &ts11);
211 
212  fr = Get_Force(pG, curG->x1, curG->x2, curG->x3,
213  curG->v1, curG->v2, curG->v3);
214 
215  fc.x1 = fd.x1+fr.x1;
216  fc.x2 = fd.x2+fr.x2;
217  fc.x3 = fd.x3+fr.x3;
218 
219 /* step 3: calculate the force at the predicted positoin */
220  fd = Get_Drag(pG, curG->property, x1n, x2n, x3n,
221  curG->v1, curG->v2, curG->v3, cell1, &ts12);
222 
223  fr = Get_Force(pG, x1n, x2n, x3n, curG->v1, curG->v2, curG->v3);
224 
225  fp.x1 = fd.x1+fr.x1;
226  fp.x2 = fd.x2+fr.x2;
227  fp.x3 = fd.x3+fr.x3;
228 
229 /* step 4: calculate the velocity update */
230  /* shortcut expressions */
231  b0 = 1.0+pG->dt*ts11;
232 
233  /* Total force */
234  ft.x1 = 0.5*(fc.x1+b0*fp.x1);
235  ft.x2 = 0.5*(fc.x2+b0*fp.x2);
236  ft.x3 = 0.5*(fc.x3+b0*fp.x3);
237 
238 #ifdef SHEARING_BOX
239  oh = Omega_0*pG->dt;
240  if (ShBoxCoord == xy) {/* (x1,x2,x3)=(X,Y,Z) */
241  ft.x1 += -oh*fp.x2;
242  #ifdef FARGO
243  ft.x2 += 0.5*(2.0-qshear)*oh*fp.x1;
244  #else
245  ft.x2 += oh*fp.x1;
246  #endif
247  } else { /* (x1,x2,x3)=(X,Z,Y) */
248  ft.x1 += -oh*fp.x3;
249  #ifdef FARGO
250  ft.x3 += 0.5*(2.0-qshear)*oh*fp.x1;
251  #else
252  ft.x3 += oh*fp.x1;
253  #endif
254  }
255 #endif /* SHEARING_BOX */
256 
257  /* calculate the inverse matrix elements */
258  D = 1.0+0.5*pG->dt*(ts11 + ts12 + pG->dt*ts11*ts12);
259 #ifdef SHEARING_BOX
260  oh2 = SQR(oh);
261  B = oh * (-2.0-(ts11+ts12)*pG->dt);
262 #ifdef FARGO
263  A = D - (2.0-qshear)*oh2;
264  C = 0.5*(qshear-2.0)*B;
265 #else /* FARGO */
266  A = D - 2.0*oh2;
267  C = -B;
268 #endif /* FARGO */
269  Det1 = 1.0/(SQR(A)-B*C);
270  if (ShBoxCoord == xy) {
271  *dv1 = pG->dt*Det1*(ft.x1*A-ft.x2*B);
272  *dv2 = pG->dt*Det1*(-ft.x1*C+ft.x2*A);
273  *dv3 = pG->dt*ft.x3/D;
274  } else {
275  *dv1 = pG->dt*Det1*(ft.x1*A-ft.x3*B);
276  *dv3 = pG->dt*Det1*(-ft.x1*C+ft.x3*A);
277  *dv2 = pG->dt*ft.x2/D;
278  }
279 #else /* SHEARING_BOX */
280  D = 1.0/D;
281  *dv1 = pG->dt*ft.x1*D;
282  *dv2 = pG->dt*ft.x2*D;
283  *dv3 = pG->dt*ft.x3*D;
284 #endif /* SHEARING_BOX */
285 
286  *ts = 0.5/ts11+0.5/ts12;
287 
288  return;
289 }

467 void feedback_predictor(DomainS *pD)
468 {
469  GridS *pG = pD->Grid;
470  int is,js,ks,i,j,k;
471  long p; /* particle index */
472  Real weight[3][3][3]; /* weight function */
473  Real rho, cs, tstop; /* density, sound speed, stopping time */
474  Real u1, u2, u3;
475  Real vd1, vd2, vd3, vd; /* velocity difference between particle and gas */
476  Real f1, f2, f3; /* feedback force */
477  Real m, ts1h; /* grain mass, 0.5*dt/tstop */
478  Real3Vect cell1; /* one over dx1, dx2, dx3 */
479  Real3Vect fb; /* drag force, fluid velocity */
480 #ifndef BAROTROPIC
481  Real Elosspar; /* energy dissipation rate due to drag */
482 #endif
483  Real stiffness; /* stiffness parameter of feedback */
484  GrainS *gr; /* pointer of the current working position */
485 
486  /* initialization */
487  get_gasinfo(pG); /* calculate gas information */
488 
489  for (k=klp; k<=kup; k++)
490  for (j=jlp; j<=jup; j++)
491  for (i=ilp; i<=iup; i++) {
492  /* clean the feedback array */
493  pG->Coup[k][j][i].fb1 = 0.0;
494  pG->Coup[k][j][i].fb2 = 0.0;
495  pG->Coup[k][j][i].fb3 = 0.0;
496 #ifndef BAROTROPIC
497  pG->Coup[k][j][i].Eloss = 0.0;
498 #endif
499  pG->Coup[k][j][i].FBstiff = 0.0;
500  }
501 
502  /* convenient expressions */
503  if (pG->Nx[0] > 1) cell1.x1 = 1.0/pG->dx1;
504  else cell1.x1 = 0.0;
505 
506  if (pG->Nx[1] > 1) cell1.x2 = 1.0/pG->dx2;
507  else cell1.x2 = 0.0;
508 
509  if (pG->Nx[2] > 1) cell1.x3 = 1.0/pG->dx3;
510  else cell1.x3 = 0.0;
511 
512  /* loop over all particles to calculate the drag force */
513  for (p=0; p<pG->nparticle; p++)
514  {/* loop over all particle */
515  gr = &(pG->particle[p]);
516 
517  /* interpolation to get fluid density and velocity */
518  getweight(pG, gr->x1, gr->x2, gr->x3, cell1, weight, &is, &js, &ks);
519  if (getvalues(pG, weight, is, js, ks,
520  &rho, &u1, &u2, &u3, &cs, &stiffness) == 0)
521  { /* particle is in the grid */
522 
523  /* apply gas velocity shift due to pressure gradient */
524  gasvshift(gr->x1, gr->x2, gr->x3, &u1, &u2, &u3);
525  /* velocity difference */
526  vd1 = u1 - gr->v1;
527  vd2 = u2 - gr->v2;
528  vd3 = u3 - gr->v3;
529  vd = sqrt(vd1*vd1 + vd2*vd2 + vd3*vd3);
530 
531  /* calculate particle stopping time */
532  tstop = get_ts(pG, gr->property, rho, cs, vd);
533  tstop = MAX(tstop, pG->dt);
534  ts1h = 0.5*pG->dt/tstop;
535 
536  /* Drag force density */
537  m = grproperty[gr->property].m;
538  fb.x1 = m * vd1 * ts1h;
539  fb.x2 = m * vd2 * ts1h;
540  fb.x3 = m * vd3 * ts1h;
541 
542  /* calculate feedback stiffness */
543  stiffness = m*pG->dt/tstop;
544 
545  /* distribute the drag force (density) to the grid */
546 #ifndef BAROTROPIC
547  Elosspar = fb.x1*vd1 + fb.x2*vd2 + fb.x3*vd3;
548  distrFB_pred(pG, weight, is, js, ks, fb, stiffness, Elosspar);
549 #else
550  distrFB_pred(pG, weight, is, js, ks, fb, stiffness);
551 #endif
552  }
553  }/* end of the for loop */
554 
555 /* normalize stiffness and correct for feedback */
556  for (k=klp; k<=kup; k++)
557  for (j=jlp; j<=jup; j++)
558  for (i=ilp; i<=iup; i++)
559  {
560  pG->Coup[k][j][i].FBstiff /= pG->U[k][j][i].d;
561 
562  stiffness = 1.0/MAX(1.0, pG->Coup[k][j][i].FBstiff);
563 
564  pG->Coup[k][j][i].fb1 *= stiffness;
565  pG->Coup[k][j][i].fb2 *= stiffness;
566  pG->Coup[k][j][i].fb3 *= stiffness;
567  }
568 
569  return;
570 }

583 void feedback_corrector(GridS *pG, GrainS *gri, GrainS *grf, Real3Vect cell1,
584  Real dv1, Real dv2, Real dv3, Real ts)

595  x2 = 0.5*(gri->x2+grf->x2);
596  x3 = 0.5*(gri->x3+grf->x3);

603 
604  fb.x1 = dv1 - pG->dt*fb.x1;
605  fb.x2 = dv2 - pG->dt*fb.x2;
606  fb.x3 = dv3 - pG->dt*fb.x3;
607 
608  /* energy dissipation */
609  Elosspar = mgr*(SQR(fb.x1)+SQR(fb.x2)+SQR(fb.x3))*(ts/pG->dt);
610 
611  /* Drag force density */
612  fb.x1 = mgr*fb.x1;
613  fb.x2 = mgr*fb.x2;
614  fb.x3 = mgr*fb.x3;
615 
616  /* distribute the drag force (density) to the grid */
617  getweight(pG, x1, x2, x3, cell1, weight, &is, &js, &ks);
618  distrFB_corr(pG, weight, is, js, ks, fb, Elosspar);

690 Real3Vect Get_Drag(GridS *pG, int type, Real x1, Real x2, Real x3,
691  Real v1, Real v2, Real v3, Real3Vect cell1, Real *tstop1)
692 {
693  int is, js, ks;
694  Real rho, u1, u2, u3, cs;
695  Real vd1, vd2, vd3, vd, tstop, ts1;
696 #ifdef FEEDBACK
697  Real stiffness;
698 #endif
699  Real weight[3][3][3]; /* weight function */
700  Real3Vect fd;
701 
702  /* interpolation to get fluid density, velocity and the sound speed */
703  getweight(pG, x1, x2, x3, cell1, weight, &is, &js, &ks);
704 
705 #ifndef FEEDBACK
706  if (getvalues(pG, weight, is, js, ks, &rho, &u1, &u2, &u3, &cs) == 0)
707 #else
708  if (getvalues(pG, weight, is, js, ks, &rho,&u1,&u2,&u3,&cs, &stiffness) == 0)
709 #endif
710  { /* particle in the grid */
711 
712  /* apply possible gas velocity shift (e.g., for fake gas velocity field) */
713  gasvshift(x1, x2, x3, &u1, &u2, &u3);
714 
715  /* velocity difference */
716  vd1 = v1-u1;
717  vd2 = v2-u2;
718  vd3 = v3-u3;
719  vd = sqrt(SQR(vd1) + SQR(vd2) + SQR(vd3)); /* dimension independent */
720 
721  /* particle stopping time */
722  tstop = get_ts(pG, type, rho, cs, vd);
723 #ifdef FEEDBACK
724  tstop *= MAX(1.0, stiffness);
725 #endif
726  ts1 = 1.0/tstop;
727  }
728  else
729  { /* particle out of the grid, free motion, with warning sign */
730  vd1 = 0.0; vd2 = 0.0; vd3 = 0.0; ts1 = 0.0;
731  ath_perr(0, "Particle move out of grid %d with position (%f,%f,%f)!\n",
732  myID_Comm_world,x1,x2,x3); /* warning! */
733  }
734 
735  *tstop1 = ts1;
736 
737  /* Drag force */
738  fd.x1 = -ts1*vd1;
739  fd.x2 = -ts1*vd2;
740  fd.x3 = -ts1*vd3;
741 
742  return fd;
743 }

757  Real v1, Real v2, Real v3)

769 #ifdef SHEARING_BOX

786  ft.x3 += (qshear-2.0)*v1*Omega_0;
787  #else

789  ft.x3 += -2.0*v1*Omega_0;
790  #endif /* FARGO */

792 #endif /* SHEARING_BOX */
