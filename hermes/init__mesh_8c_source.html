895 static int dom_decomp_2d(const int Nx, const int Ny,
896  const int Np, int *pNGx, int *pNGy){
897 
898  int rx, ry, I, rxs, rx_min, rx_max;
899  int rx0=1, ry0=1, I0=0, init=1;
900  div_t dv;
901 
902  /* Compute the ideal decomposition, truncated to an integer, which
903  minimizes the amount of communication. */
904  rxs = (int)sqrt((double)(Nx*Np)/(double)(Ny > 2*nghost ? Ny - 2*nghost : 1));
905 
906  /* Constrain the decomposition */
907  rx_max = Nx < Np ? Nx : Np; /* Require ry >= 1 and rx <= Nx */
908  if(Ny < Np){ /* Require rx >= 1 and ry <= Ny */
909  dv = div(Np, Ny);
910  /* rx_min = the smallest integer >= Np/Ny */
911  rx_min = dv.quot + (dv.rem > 0 ? 1 : 0);
912  }
913  else rx_min = 1;
914 
915  /* Constrain rxs to fall in this domain */
916  rxs = rxs > rx_min ? rxs : rx_min;
917  rxs = rxs < rx_max ? rxs : rx_max;
918 
919  /* Search down for a factor of Np */
920  for(rx=rxs; rx>=rx_min; rx--){
921  dv = div(Np,rx);
922  if(dv.rem == 0){
923  rx0 = rx;
924  ry0 = dv.quot;
925  I0 = (rx0 - 1)*Ny + (ry0 - 1)*(Nx + 2*nghost*rx0);
926  init = 0;
927  break;
928  }
929  }
930 
931  /* Search up for a factor of Np */
932  for(rx=rxs+1; rx<=rx_max; rx++){
933  dv = div(Np,rx);
934  if(dv.rem == 0){
935  ry = dv.quot;
936  I = (rx - 1)*Ny + (ry - 1)*(Nx + 2*nghost*rx);
937 
938  if(init || I < I0){
939  rx0 = rx;
940  ry0 = ry;
941  I0 = I;
942  init = 0;
943  }
944  break;
945  }
946  }
947 
948  if(init) return 1; /* Error locating a solution */
949 
950  *pNGx = rx0;
951  *pNGy = ry0;
952 
953  return 0;
954 }

964 static int dom_decomp_3d(const int Nx, const int Ny, const int Nz,
965  const int Np, int *pNGx, int *pNGy, int *pNGz){

972  /* Constrain the decomposition */

1004  }
1005  }

1007  }
