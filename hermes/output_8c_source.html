481 {

497  OutArray[n].t += OutArray[n].dt;
498  dump_flag[n] = 1;

507  rst_out.t += rst_out.dt;

513  for(n=0; n<out_count; n++){

515  if(OutArray[n].n > 0){
516  sprintf(block,"output%d",OutArray[n].n);
517  if (dump_flag[n] != 0) {
518 /* About to write this output, so increase the output
519  * number given in the restart file */
520  par_seti(block,"num","%d",OutArray[n].num+1,"Next Output Number");
521  } else {
522  par_seti(block,"num","%d",OutArray[n].num,"Next Output Number");
523  }

525  }
526  }

528  sprintf(block,"output%d",rst_out.n);
529  par_seti(block,"num","%d",rst_out.num+1,"Next Output Number");
530  par_setd(block,"time","%.15e",rst_out.t,"Next Output Time");
531 
532 /* Write the restart file */
533  (*(rst_out.res_fun))(pM,&(rst_out));
534 
535  rst_out.num++;
536  }
537  }
538 
539 /* Loop over all elements in output array, if dump_flag != 0, make output */
540 
541  for (n=0; n<out_count; n++) {
542  if(dump_flag[n] != 0) {
543 
544 #ifdef PARTICLES
545  if (OutArray[n].out_pargrid == 1) /* binned particles are output */
546  if (OutArray[n].par_prop != mypar_prop) {
547  particle_to_grid(pD, OutArray[n].par_prop);
548  mypar_prop = OutArray[n].par_prop;
549  }
550 #endif
551  (*OutArray[n].out_fun)(pM,&(OutArray[n]));
552 
553  OutArray[n].num++;
554 
555  }
556  }
557 
558  return;
559 }

593  }
594

596  }
597  if (OutArray[i].out_fmt != NULL) free(OutArray[i].out_fmt);

649  }

653  ku = pgrid->ke + nghost;
654  } else{
655  kl = pgrid->ks;

671  if (data == NULL) ath_error("[OutData3] Error creating 3D data array\n");
672  for (k=0; k<*Nx3; k++)
673  for (j=0; j<*Nx2; j++)
674  for (i=0; i<*Nx1; i++)
675  data[k][j][i] = (*pout->expr)(pgrid,i+il,j+jl,k+kl);
