87  return;

91  dat += 2;
92  }

94  while (cnt--) {
95  tmp = dat[0]; dat[0] = dat[3]; dat[3] = tmp;
96  tmp = dat[1]; dat[1] = dat[2]; dat[2] = tmp;
97  dat += 4;
98  }
99  else if (len==8)
100  while (cnt--) {
101  tmp = dat[0]; dat[0] = dat[7]; dat[7] = tmp;
102  tmp = dat[1]; dat[1] = dat[6]; dat[6] = tmp;
103  tmp = dat[2]; dat[2] = dat[5]; dat[5] = tmp;
104  tmp = dat[3]; dat[3] = dat[4]; dat[4] = tmp;
105  dat += 8;
106  }
107  else { /* the general SLOOOOOOOOOW case */
108  for(k=0; k<len/2; k++) {
109  tmp = dat[k];
110  dat[k] = dat[len-1-k];
111  dat[len-1-k] = tmp;
112  }
113  }

405 void compute_l1_error(const char *problem, const MeshS *pM,
406  const ConsS ***RootSoln, const int errortype)
407 {
408  DomainS *pD=&(pM->Domain[0][0]);
409  GridS *pG=pM->Domain[0][0].Grid;
410  int i=0,j=0,k=0;
411 #if (NSCALARS > 0)
412  int n;
413 #endif
414  int is,ie,js,je,ks,ke;
415  Real rms_error=0.0;
416  Real dVol,totVol;
417  ConsS error,total_error;
418  FILE *fp;
419  char *fname, fnamestr[256];
420  int Nx1,Nx2,Nx3;
421 #if defined MPI_PARALLEL
422  double err[8+NSCALARS], tot_err[8+NSCALARS];
423  int mpi_err;
424 #endif
425 #ifdef CYLINDRICAL
426  Real x1,x2,x3;
427 #endif
428 
429  /* Clear out the total_error struct */
430  memset(&total_error,0.0,sizeof(ConsS));
431  if (pG == NULL) return;
432 
433 /* compute L1 error in each variable, and rms total error */
434 
435  is = pG->is; ie = pG->ie;
436  js = pG->js; je = pG->je;
437  ks = pG->ks; ke = pG->ke;
438 
439  for (k=ks; k<=ke; k++) {
440  for (j=js; j<=je; j++) {
441  memset(&error,0.0,sizeof(ConsS));
442  for (i=is; i<=ie; i++) {
443  dVol = 1.0;
444  if (pG->dx1 > 0.0) dVol *= pG->dx1;
445  if (pG->dx2 > 0.0) dVol *= pG->dx2;
446  if (pG->dx3 > 0.0) dVol *= pG->dx3;
447 #ifdef CYLINDRICAL
448  cc_pos(pG,i,j,k,&x1,&x2,&x3);
449  dVol *= x1;
450 #endif
451 
452  /* Sum local L1 error for each grid cell I own */
453  error.d += dVol*fabs(pG->U[k][j][i].d - RootSoln[k][j][i].d );
454  error.M1 += dVol*fabs(pG->U[k][j][i].M1 - RootSoln[k][j][i].M1);
455  error.M2 += dVol*fabs(pG->U[k][j][i].M2 - RootSoln[k][j][i].M2);
456  error.M3 += dVol*fabs(pG->U[k][j][i].M3 - RootSoln[k][j][i].M3);
457 #ifdef MHD
458  error.B1c += dVol*fabs(pG->U[k][j][i].B1c - RootSoln[k][j][i].B1c);
459  error.B2c += dVol*fabs(pG->U[k][j][i].B2c - RootSoln[k][j][i].B2c);
460  error.B3c += dVol*fabs(pG->U[k][j][i].B3c - RootSoln[k][j][i].B3c);
461 #endif /* MHD */
462 #ifndef ISOTHERMAL
463  error.E += dVol*fabs(pG->U[k][j][i].E - RootSoln[k][j][i].E );
464 #endif /* ISOTHERMAL */
465 #if (NSCALARS > 0)
466  for (n=0; n<NSCALARS; n++)
467  error.s[n] += dVol*fabs(pG->U[k][j][i].s[n] - RootSoln[k][j][i].s[n]);;
468 #endif
469  }
470 
471  /* total_error is sum of local L1 error */
472  total_error.d += error.d;
473  total_error.M1 += error.M1;
474  total_error.M2 += error.M2;
475  total_error.M3 += error.M3;
476 #ifdef MHD
477  total_error.B1c += error.B1c;
478  total_error.B2c += error.B2c;
479  total_error.B3c += error.B3c;
480 #endif /* MHD */
481 #ifndef ISOTHERMAL
482  total_error.E += error.E;
483 #endif /* ISOTHERMAL */
484 #if (NSCALARS > 0)
485  for (n=0; n<NSCALARS; n++) total_error.s[n] += error.s[n];
486 #endif
487  }
488  }
489 
490 #ifdef MPI_PARALLEL
491 /* Now we have to use an All_Reduce to get the total error over all the MPI
492  * grids. Begin by copying the error into the err[] array */
493 
494  err[0] = total_error.d;
495  err[1] = total_error.M1;
496  err[2] = total_error.M2;
497  err[3] = total_error.M3;
498 #ifdef MHD
499  err[4] = total_error.B1c;
500  err[5] = total_error.B2c;
501  err[6] = total_error.B3c;
502 #endif /* MHD */
503 #ifndef ISOTHERMAL
504  err[7] = total_error.E;
505 #endif /* ISOTHERMAL */
506 #if (NSCALARS > 0)
507  for (n=0; n<NSCALARS; n++) err[8+n] = total_error.s[n];
508 #endif
509 
510 /* Sum up the Computed Error */
511  mpi_err = MPI_Reduce(err, tot_err, (8+NSCALARS),
512  MPI_DOUBLE, MPI_SUM, 0, pD->Comm_Domain);
513  if(mpi_err)
514  ath_error("[compute_l1_error]: MPI_Reduce call returned error = %d\n",
515  mpi_err);
516 
517 /* If I'm the parent, copy the sum back to the total_error variable */
518  if(pD->DomNumber == 0){ /* I'm the parent */
519  total_error.d = tot_err[0];
520  total_error.M1 = tot_err[1];
521  total_error.M2 = tot_err[2];
522  total_error.M3 = tot_err[3];
523 #ifdef MHD
524  total_error.B1c = tot_err[4];
525  total_error.B2c = tot_err[5];
526  total_error.B3c = tot_err[6];
527 #endif /* MHD */
528 #ifndef ISOTHERMAL
529  total_error.E = tot_err[7];
530 #endif /* ISOTHERMAL */
531 #if (NSCALARS > 0)
532  for (n=0; n<NSCALARS; n++) total_error.s[n] = err[8+n];
533 #endif
534 
535  }
536  else return; /* The child grids do not do any of the following code */
537 #endif /* MPI_PARALLEL */
538 
539  /* Compute total number of grid cells */
540  Nx1 = pD->Nx[0];
541  Nx2 = pD->Nx[1];
542  Nx3 = pD->Nx[2];
543 
544  totVol = 1.0;
545  if (errortype == 1) {
546  if (pD->MaxX[0] > pD->MinX[0]) totVol *= pD->MaxX[0] - pD->MinX[0];
547  if (pD->MaxX[1] > pD->MinX[1]) totVol *= pD->MaxX[1] - pD->MinX[1];
548  if (pD->MaxX[2] > pD->MinX[2]) totVol *= pD->MaxX[2] - pD->MinX[2];
549 #ifdef CYLINDRICAL
550  totVol *= 0.5*(pD->MinX[0] + pD->MaxX[0]);
551 #endif
552  }
553 
554 
555 /* Compute RMS error over all variables, and print out */
556 
557  rms_error = SQR(total_error.d) + SQR(total_error.M1) + SQR(total_error.M2)
558  + SQR(total_error.M3);
559 #ifdef MHD
560  rms_error += SQR(total_error.B1c) + SQR(total_error.B2c)
561  + SQR(total_error.B3c);
562 #endif /* MHD */
563 #ifndef ISOTHERMAL
564  rms_error += SQR(total_error.E);
565 #endif /* ISOTHERMAL */
566 
567  rms_error = sqrt(rms_error)/totVol;
568 
569 /* Print error to file "BLAH-errors.#.dat" */
570  sprintf(fnamestr,"%s-errors",problem);
571  fname = ath_fname(NULL,fnamestr,NULL,NULL,1,0,NULL,"dat");
572 
573 /* The file exists -- reopen the file in append mode */
574  if((fp=fopen(fname,"r")) != NULL){
575  if((fp = freopen(fname,"a",fp)) == NULL){
576  ath_error("[compute_l1_error]: Unable to reopen file.\n");
577  free(fname);
578  return;
579  }
580  }
581 /* The file does not exist -- open the file in write mode */
582  else{
583  if((fp = fopen(fname,"w")) == NULL){
584  ath_error("[compute_l1_error]: Unable to open file.\n");
585  free(fname);
586  return;
587  }
588 /* Now write out some header information */
589  fprintf(fp,"# Nx1 Nx2 Nx3 RMS-Error d M1 M2 M3");

591  fprintf(fp," E");
592 #endif /* ISOTHERMAL */
593 #ifdef MHD

595 #endif /* MHD */
596 #if (NSCALARS > 0)
597  for (n=0; n<NSCALARS; n++) {
598  fprintf(fp," S[ %d ]",n);
599  }
600 #endif
601  fprintf(fp,"\n#\n");
602  }
603 
604  fprintf(fp,"%d %d %d %e",Nx1,Nx2,Nx3,rms_error);
605 
606  fprintf(fp," %e %e %e %e",
607  (total_error.d /totVol),
608  (total_error.M1/totVol),
609  (total_error.M2/totVol),
610  (total_error.M3/totVol));
611 
612 #ifndef ISOTHERMAL
613  fprintf(fp," %e",total_error.E/totVol);
614 #endif /* ISOTHERMAL */
615

618  (total_error.B1c/totVol),
619  (total_error.B2c/totVol),
620  (total_error.B3c/totVol));
621 #endif /* MHD */
