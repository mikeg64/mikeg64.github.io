83  nx3 = (ke-ks)+1 + 2*nghost;

85  ath_error("[pgflow]: Error allocating memory\n");

87 /* An angle = 0.0 is a wave aligned with the x1-direction. */
88  angle = par_getd("problem","angle");
89 
90 /* If the grid is 1-D we override the angle variable */
91  if(pGrid->Nx[1] <= 1) angle = 0.0;
92  if(pGrid->Nx[0] <= 1) angle = 90.0;
93 
94 /* Compute the sin and cos of the angle and the wavelength. */
95  if (angle == 0.0) {
96  sin_a = 0.0;
97  cos_a = 1.0;
98  lambda = pGrid->Nx[0]*pGrid->dx1; /* Put one wavelength in the grid */
99  }
100  else if (angle == 90.0) {
101  sin_a = 1.0;
102  cos_a = 0.0;
103  lambda = pGrid->Nx[1]*pGrid->dx2; /* Put one wavelength in the grid */
104  }
105  else {
106 /* We put 1 wavelength in each direction. Hence the wavelength
107  * lambda = pGrid->Nx[0]*pGrid->dx1*cos_a;
108  * AND lambda = pGrid->Nx[1]*pGrid->dx2*sin_a;
109  * are both satisfied. */
110  if((pGrid->Nx[0]*pGrid->dx1) == (pGrid->Nx[1]*pGrid->dx2)){
111  cos_a = sin_a = sqrt(0.5);
112  }
113  else{
114  angle = atan((double)(pGrid->Nx[0]*pGrid->dx1)/(pGrid->Nx[1]*pGrid->dx2));
115  sin_a = sin(angle);
116  cos_a = cos(angle);
117  }
118 /* Use the larger angle to determine the wavelength */
119  if (cos_a >= sin_a) {
120  lambda = pGrid->Nx[0]*pGrid->dx1*cos_a;
121  } else {
122  lambda = pGrid->Nx[1]*pGrid->dx2*sin_a;
123  }
124  }
125 
126 /* Initialize k_parallel */
127  k_par = 2.0*PI/lambda;
128 
129  E0 = 0.0; /* Initialize the total energy */
130  grav = par_getd("problem","grav");
131  root = par_geti("problem","root");
132

134  pres = par_getd("problem","pres");
135  v_par = par_getd("problem","v_par");
136 
137 /* Set up the constants of motion */
138  Phi = den*v_par; /* The mass flux */
139  S = pres/pow((double)den,(double)Gamma); /* specific entropy */
140 /* Calculate the Bernoulli constant at x1 = 0.0 */
141  H = 0.5*v_par*v_par + Gamma*pres/(Gamma_1*den);
142 
143  rho_e = pow((double)(Phi*Phi/(Gamma*S)),(double)(1.0/(Gamma+1.0)));
144 
145  for (k=ks; k<=ke; k++) {
146  for (j=js; j<=je; j++) {
147  for (i=is; i<=ie; i++) {
148  cc_pos(pGrid,i,j,k,&x1,&x2,&x3);
149 
150 /* Calculate the gravitational potential */
151  psi = -grav*sin((double)k_par*(x1*cos_a + x2*sin_a))/k_par;
152 
153  if(H <= psi)
154  ath_error("[problem]: H < Psi -- No solution exists\n");
155 
156  if(Bfunc((double)rho_e) < 0.0)
157  ath_error("[problem]: Bfunc(rho_e) < 0.0 -- No solution exists\n");
158 
159  if(root){ /* Choose the heavy (subsonic) root */
160 /* The root is bounded: rho_e <= rho < rho_s */
161  rho_s = pow((double)(Gamma_1*(H-psi)/(Gamma*S)),(double)(1.0/Gamma_1));
162 
163  if(rtbis(Bfunc,rho_e,rho_s,1.0e-12*rho_e,100,&rho)){
164  exit(1);
165  }
166  } else { /* Choose the light (supersonic) root */
167 /* The root is bounded: rho_p < rho <= rho_e */
168  rho_p = fabs((double)Phi)/sqrt((double)(2.0*(H-psi)));
169 
170  if(rtbis(Bfunc,rho_p,rho_e,1.0e-12*rho_e,100,&rho)){
171  exit(1);
172  }
173  }
174 
175  d0[k][j][i] = rho;
176  pGrid->U[k][j][i].d = rho;
177  pGrid->U[k][j][i].M1 = Phi*cos_a;
178  pGrid->U[k][j][i].M2 = Phi*sin_a;
179  pGrid->U[k][j][i].M3 = 0.0;
180  pGrid->U[k][j][i].E = S*pow(rho,(double)Gamma)/Gamma_1
181  + 0.5*Phi*Phi/rho;
182 
183  E0 += pGrid->U[k][j][i].E + rho*psi;
184  }
185  }}

188  E0 /= (Real)((ie - is + 1)*(je - js + 1)*(ke - ks + 1));
189

191  StaticGravPot = grav_pot;

266 int rtbis(double (*pfun)(double), const double x1, const double x2,
267  const double xacc, const int imax, double *prt)
268 {
269  int i;
270  double xn, xm, xp, dx;
271  double fn = (*pfun)(x1), fm, fp = (*pfun)(x2);
272 
273  if(fn < 0.0 && fp > 0.0){
274  xn = x1;
275  xp = x2;
276  }
277  else if(fn > 0.0 && fp < 0.0){
278  xn = x2;
279  xp = x1;
280 
281  fm = fn;
282  fn = fp;
283  fp = fm;
284  }
285  else if(fn == 0.0){ *prt = x1; return 0; }
286  else if(fp == 0.0){ *prt = x2; return 0; }
287  else{
288  ath_perr(-1,"[rtbis]: Root must be bracketed for bisection\n");
289  ath_perr(-1,"[rtbis]: x1=%g, x2=%g, F(x1)=%g, F(x2)=%g\n",
290  x1,x2,fn,fp);
291  return 1; /* Error */
292  }
293 
294  dx = xp - xn;
295 
296  for(i=0; i<imax; i++){ /* Bisection loop */
297  dx *= 0.5;
298  xm = xn + dx;
299  fm = (*pfun)(xm);
300 
301  if(fm < 0.0){ xn = xm; fn = fm; }
302  else { xp = xm; fp = fm; }
303 
304  if(fabs(dx) < xacc || fm == 0.0){
305  *prt = fp < -fn ? xp : xn; /* Return our best value for the root */
306  return 0;
307  }
308  }
309 
310  ath_perr(-1,"[rtbis]: Too many bisections\n");
311 
312  *prt = fp < -fn ? xp : xn; /* Return our best value for the root */
313 
314  return 1;
315 }
