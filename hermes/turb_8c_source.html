121 /*
122  * Function pspect -- computes component of velocity with specific power
123  * spectrum in Fourier space determined by ispect
124  *
125  * Velocity power spectrum returned in ampl
126  * klow = multiple of 2 pi/L for cut-off at low wavenumbers
127  * khigh = multiple of 2 pi/L for cut-off at high wavenumbers
128  * expo = exponent of power law
129  * ispect = integer flag which specifies spectrum
130  * Note that the fourier amplitudes are stored in an array with no
131  * ghost zones
132  */
133 
134 static void pspect(ath_fft_data *ampl)
135 {
136  int i,j,k;
137  double q1,q2,q3;
138 
139  /* set random amplitudes with gaussian deviation */
140  for (k=0; k<nx3; k++) {
141  for (j=0; j<nx2; j++) {
142  for (i=0; i<nx1; i++) {
143  q1 = ran2(&rseed);
144  q2 = ran2(&rseed);
145  q3 = sqrt(-2.0*log(q1+1.0e-20))*cos(2.0*PI*q2);
146  q1 = ran2(&rseed);
147  ampl[OFST(i,j,k)][0] = q3*cos(2.0*PI*q1);
148  ampl[OFST(i,j,k)][1] = q3*sin(2.0*PI*q1);
149  }
150  }
151  }
152 
153  /* set power spectrum
154  * ispect=1: power law - original form
155  * ispect=2: form from Gammie&Ostriker
156  */
157  for (k=0; k<nx3; k++) {
158  for (j=0; j<nx2; j++) {
159  for (i=0; i<nx1; i++) {
160  /* compute k/dkx */
161  q3 = KWVM(i,j,k);
162  if ((q3 > klow) && (q3 < khigh)) {
163  q3 *= dkx; /* multiply by 2 pi/L */
164  if (ispect == 1) {
165  /* decreasing power law */
166  ampl[OFST(i,j,k)][0] /= pow(q3,(expo+2.0)/2.0);
167  ampl[OFST(i,j,k)][1] /= pow(q3,(expo+2.0)/2.0);
168  } else if (ispect == 2) {
169  /* G&O form */
170  ampl[OFST(i,j,k)][0] *= pow(q3,3.0)*exp(-4.0*q3/kpeak);
171  ampl[OFST(i,j,k)][1] *= pow(q3,3.0)*exp(-4.0*q3/kpeak);
172  }
173  } else {
174  /* introduce cut-offs at klow and khigh */
175  ampl[OFST(i,j,k)][0] = 0.0;
176  ampl[OFST(i,j,k)][1] = 0.0;
177  }
178  }
179  }
180  }
181  ampl[0][0] = 0.0;
182  ampl[0][1] = 0.0;
183 
184  return;
185 }

189 /*
190  * Function project
191  *
192  * Makes velocity perturbations divergence free
193  */
194 
195 static void project()
196 {
197  int i,j,k,m,ind;
198  double kap[3], kapn[3], mag;
199  ath_fft_data dot;
200 
201  /* Project off non-solenoidal component of velocity */
202  for (k=0; k<nx3; k++) {
203  kap[2] = sin(2.0*PI*(gks+k)/gnx3);
204  for (j=0; j<nx2; j++) {
205  kap[1] = sin(2.0*PI*(gjs+j)/gnx2);
206  for (i=0; i<nx1; i++) {
207  if (((gis+i)+(gjs+j)+(gks+k)) != 0) {
208  kap[0] = sin(2.0*PI*(gis+i)/gnx1);
209  ind = OFST(i,j,k);
210 
211  /* make kapn a unit vector */
212  mag = sqrt(SQR(kap[0]) + SQR(kap[1]) + SQR(kap[2]));
213  for (m=0; m<3; m++) kapn[m] = kap[m] / mag;
214 
215  /* find fv_0 dot kapn */
216  dot[0] = fv1[ind][0]*kapn[0]+fv2[ind][0]*kapn[1]+fv3[ind][0]*kapn[2];
217  dot[1] = fv1[ind][1]*kapn[0]+fv2[ind][1]*kapn[1]+fv3[ind][1]*kapn[2];
218 
219  /* fv = fv_0 - (fv_0 dot kapn) * kapn */
220  fv1[ind][0] -= dot[0]*kapn[0];
221  fv2[ind][0] -= dot[0]*kapn[1];
222  fv3[ind][0] -= dot[0]*kapn[2];
223 
224  fv1[ind][1] -= dot[1]*kapn[0];
225  fv2[ind][1] -= dot[1]*kapn[1];
226  fv3[ind][1] -= dot[1]*kapn[2];
227  }
228  }
229  }
230  }
231 
232  return;
233 }

284 /*
285  * Function perturb
286  *
287  * Shifts velocities so no net momentum change, normalizes to keep
288  * dedt fixed, and then sets velocities
289  */
290 
291 static void perturb(Grid *pGrid, Real dt)
292 {
293  int i, is=pGrid->is, ie = pGrid->ie;
294  int j, js=pGrid->js, je = pGrid->je;
295  int k, ks=pGrid->ks, ke = pGrid->ke;
296  int ind, mpierr;
297  Real dvol, aa, b, c, s, de, qa, v1, v2, v3;
298  Real t0, t0ij, t0i, t1, t1ij, t1i;
299  Real t2, t2ij, t2i, t3, t3ij, t3i;
300  Real m[4], gm[4];
301 
302  /* Set the velocities in real space */
303  dvol = 1.0/((Real)(gnx1*gnx2*gnx3));
304  for (k=ks; k<=ke; k++) {
305  for (j=js; j<=je; j++) {
306  for (i=is; i<=ie; i++) {
307  ind = OFST(i-is,j-js,k-ks);
308  dv1[k][j][i] = fv1[ind][0]*dvol;
309  dv2[k][j][i] = fv2[ind][0]*dvol;
310  dv3[k][j][i] = fv3[ind][0]*dvol;
311  }
312  }
313  }
314 
315  /* Calculate net momentum pertubation components t1, t2, t3 */
316  t0 = 0.0; t1 = 0.0; t2 = 0.0; t3 = 0.0;
317  for (k=ks; k<=ke; k++) {
318  t0ij = 0.0; t1ij = 0.0; t2ij = 0.0; t3ij = 0.0;
319  for (j=js; j<=je; j++) {
320  t0i = 0.0; t1i = 0.0; t2i = 0.0; t3i = 0.0;
321  for (i=is; i<=ie; i++) {
322  t0i += pGrid->U[k][j][i].d;
323 
324  /* The net momentum perturbation */
325  t1i += pGrid->U[k][j][i].d * dv1[k][j][i];
326  t2i += pGrid->U[k][j][i].d * dv2[k][j][i];
327  t3i += pGrid->U[k][j][i].d * dv3[k][j][i];
328  }
329  t0ij += t0i; t1ij += t1i; t2ij += t2i; t3ij += t3i;
330  }
331  t0 += t0ij; t1 += t1ij; t2 += t2ij; t3 += t3ij;
332  }
333 
334 #ifdef MPI_PARALLEL
335  /* Sum the perturbations over all processors */
336  m[0] = t0; m[1] = t1; m[2] = t2; m[3] = t3;
337  mpierr = MPI_Allreduce(m, gm, 4, MPI_RL, MPI_SUM, MPI_COMM_WORLD);
338  if (mpierr) ath_error("[normalize]: MPI_Allreduce error = %d\n", mpierr);
339  t0 = gm[0]; t1 = gm[1]; t2 = gm[2]; t3 = gm[3];
340 #endif /* MPI_PARALLEL */
341 
342  /* Subtract the mean velocity perturbation so that the net momentum
343  * perturbation is zero. */
344  for (k=ks; k<=ke; k++) {
345  for (j=js; j<=je; j++) {
346  for (i=is; i<=ie; i++) {
347  dv1[k][j][i] -= t1/t0;
348  dv2[k][j][i] -= t2/t0;
349  dv3[k][j][i] -= t3/t0;
350  }
351  }
352  }
353 
354  /* Calculate unscaled energy of perturbations */
355  t1 = 0.0; t2 = 0.0;
356  for (k=ks; k<=ke; k++) {
357  t1ij = 0.0; t2ij = 0.0;
358  for (j=js; j<=je; j++) {
359  t1i = 0.0; t2i = 0.0;
360  for (i=is; i<=ie; i++) {
361  /* Calculate velocity pertubation at cell center from
362  * perturbations at cell faces */
363  v1 = dv1[k][j][i];
364  v2 = dv2[k][j][i];
365  v3 = dv3[k][j][i];
366 
367  t1i += (pGrid->U[k][j][i].d)*(SQR(v1) + SQR(v2) + SQR(v3));
368  t2i += (pGrid->U[k][j][i].M1)*v1 + (pGrid->U[k][j][i].M2)*v2 +
369  (pGrid->U[k][j][i].M3)*v3;
370  }
371  t1ij += t1i; t2ij += t2i;
372  }
373  t1 += t1ij; t2 += t2ij;
374  }
375 
376 #ifdef MPI_PARALLEL
377  /* Sum the perturbations over all processors */
378  m[0] = t1; m[1] = t2;
379  mpierr = MPI_Allreduce(m, gm, 2, MPI_RL, MPI_SUM, MPI_COMM_WORLD);
380  if (mpierr) ath_error("[normalize]: MPI_Allreduce error = %d\n", mpierr);
381  t1 = gm[0]; t2 = gm[1];
382 #endif /* MPI_PARALLEL */
383 
384  /* Rescale to give the correct energy injection rate */
385  dvol = pGrid->dx1*pGrid->dx2*pGrid->dx3;
386  if (idrive == 0) {
387  /* driven turbulence */
388  de = dedt*dt;
389  } else {
390  /* decaying turbulence (all in one shot) */
391  de = dedt;
392  }
393  aa = 0.5*t1;
394  aa = MAX(aa,1.0e-20);
395  b = t2;
396  c = -de/dvol;
397  if(b >= 0.0)
398  s = (-2.0*c)/(b + sqrt(b*b - 4.0*aa*c));
399  else
400  s = (-b + sqrt(b*b - 4.0*aa*c))/(2.0*aa);
401 
402  if (isnan(s)) ath_error("[perturb]: s is NaN!\n");
403 
404  /* Apply momentum pertubations */
405  for (k=ks; k<=ke; k++) {
406  for (j=js; j<=je; j++) {
407  for (i=is; i<=ie; i++) {
408  qa = s*pGrid->U[k][j][i].d;
409  pGrid->U[k][j][i].M1 += qa*dv1[k][j][i];
410  pGrid->U[k][j][i].M2 += qa*dv2[k][j][i];
411  pGrid->U[k][j][i].M3 += qa*dv3[k][j][i];
412  }
413  }
414  }
415 
416  return;
417 }

435 /* -----------------------------------------------------------

437  * within preprocessor macros. Don't create variables with
438  * these names within your function if you are going to use
439  * OFST(), KCOMP(), or KWVM() within the function! */
440

449  gnx3 = pD->kde - pD->kds + 1;

452  gis=is+pGrid->idisp; gie=ie+pGrid->idisp;
453  gjs=js+pGrid->jdisp; gje=je+pGrid->jdisp;
454  gks=ks+pGrid->kdisp; gke=ke+pGrid->kdisp;

460  nx3gh = nx3 + 2*nghost;

462  /* Get input parameters */

466  dtdrive = par_getd("problem","dtdrive");
467 #ifdef MHD

469  beta = par_getd("problem","beta");

472 #endif /* MHD */

474  dedt = par_getd("problem","dedt");
475 
476  /* parameters for spectrum */
477  ispect = par_geti("problem","ispect");
478  if (ispect == 1) {
479  expo = par_getd("problem","expo");
480  } else if (ispect == 2) {

482  } else {
483  ath_error("Invalid value for ispect\n");
484  }
485  /* Cutoff wavenumbers of spectrum */
486  klow = par_getd("problem","klow"); /* in integer units */
487  khigh = par_getd("problem","khigh"); /* in integer units */
488  dkx = 2.0*PI/(pGrid->dx1*gnx1); /* convert k from integer */
489 
490  /* Driven or decaying */
491  idrive = par_geti("problem","idrive");
492  if ((idrive < 0) || (idrive > 1)) ath_error("Invalid value for idrive\n");

494  if ((idrive == 1) && (pGrid->nstep > 0)) {
495  donedrive = 1;
496  }
497 
498  if (donedrive == 0) {
499  /* Allocate memory for components of velocity perturbation */
500  if ((dv1=(Real***)calloc_3d_array(nx3gh,nx2gh,nx1gh,sizeof(Real)))==NULL) {
501  ath_error("[problem]: Error allocating memory for vel pert\n");
502  }
503  if ((dv2=(Real***)calloc_3d_array(nx3gh,nx2gh,nx1gh,sizeof(Real)))==NULL) {
504  ath_error("[problem]: Error allocating memory for vel pert\n");
505  }
506  if ((dv3=(Real***)calloc_3d_array(nx3gh,nx2gh,nx1gh,sizeof(Real)))==NULL) {
507  ath_error("[problem]: Error allocating memory for vel pert\n");
508  }
509  }
510 
511  /* Initialize the FFT plan */
512  plan = ath_3d_fft_quick_plan(pGrid, pD, NULL, ATH_FFT_BACKWARD);
513 
514  /* Allocate memory for FFTs */
515  if (donedrive == 0) {
516  fv1 = ath_3d_fft_malloc(plan);
517  fv2 = ath_3d_fft_malloc(plan);
518  fv3 = ath_3d_fft_malloc(plan);
519  }
520 
521  /* Enroll outputs */
522  dump_history_enroll(hst_dEk,"<dE_K>");
523  dump_history_enroll(hst_dEb,"<dE_B>");
524

526 }

536 void problem(Grid *pGrid, Domain *pD)
537 {
538  int i, is=pGrid->is, ie = pGrid->ie;
539  int j, js=pGrid->js, je = pGrid->je;
540  int k, ks=pGrid->ks, ke = pGrid->ke;
541 
542  rseed = (pGrid->my_id+1);
543  initialize(pGrid, pD);
544  tdrive = 0.0;
545 
546  /* Initialize uniform density and momenta */
547  for (k=ks-nghost; k<=ke+nghost; k++) {
548  for (j=js-nghost; j<=je+nghost; j++) {
549  for (i=is-nghost; i<=ie+nghost; i++) {
550  pGrid->U[k][j][i].d = rhobar;
551  pGrid->U[k][j][i].M1 = 0.0;
552  pGrid->U[k][j][i].M2 = 0.0;
553  pGrid->U[k][j][i].M3 = 0.0;
554  }
555  }
556  }
557 
558 #ifdef MHD
559  /* Initialize uniform magnetic field */
560  for (k=ks-nghost; k<=ke+nghost; k++) {
561  for (j=js-nghost; j<=je+nghost; j++) {
562  for (i=is-nghost; i<=ie+nghost; i++) {
563  pGrid->U[k][j][i].B1c = B0;
564  pGrid->U[k][j][i].B2c = 0.0;
565  pGrid->U[k][j][i].B3c = 0.0;
566  pGrid->B1i[k][j][i] = B0;
567  pGrid->B2i[k][j][i] = 0.0;
568  pGrid->B3i[k][j][i] = 0.0;
569  }
570  }
571  }
572 #endif /* MHD */
573 
574  /* Set the initial perturbations. Note that we're putting in too much
575  * energy this time. This is okay since we're only interested in the
576  * saturated state. */

724 /* The routine ran2() is extracted from the Numerical Recipes in C
725  (version 2) code. I've modified it to use doubles instead of
726  floats. -- T. A. Gardiner -- Aug. 12, 2003 */
727 
728 /* Long period (> 2 x 10^{18}) random number generator of L'Ecuyer
729  with Bays-Durham shuffle and added safeguards. Returns a uniform
730  random deviate between 0.0 and 1.0 (exclusive of the endpoint
731  values). Call with idum = a negative integer to initialize;
732  thereafter, do not alter idum between successive deviates in a
733  sequence. RNMX should appriximate the largest floating point value
734  that is less than 1. */
735 
736 double ran2(long int *idum){
737  int j;
738  long int k;
739  static long int idum2=123456789;
740  static long int iy=0;
741  static long int iv[NTAB];
742  double temp;
743 
744  if (*idum <= 0) { /* Initialize */
745  if (-(*idum) < 1) *idum=1; /* Be sure to prevent idum = 0 */
746  else *idum = -(*idum);
747  idum2=(*idum);
748  for (j=NTAB+7;j>=0;j--) { /* Load the shuffle table (after 8 warm-ups) */
749  k=(*idum)/IQ1;
750  *idum=IA1*(*idum-k*IQ1)-k*IR1;
751  if (*idum < 0) *idum += IM1;
752  if (j < NTAB) iv[j] = *idum;
753  }
754  iy=iv[0];
755  }
756  k=(*idum)/IQ1; /* Start here when not initializing */
757  *idum=IA1*(*idum-k*IQ1)-k*IR1; /* Compute idum=(IA1*idum) % IM1 without */
758  if (*idum < 0) *idum += IM1; /* overflows by Schrage's method */
759  k=idum2/IQ2;
760  idum2=IA2*(idum2-k*IQ2)-k*IR2; /* Compute idum2=(IA2*idum) % IM2 likewise */
761  if (idum2 < 0) idum2 += IM2;
762  j=(int)(iy/NDIV); /* Will be in the range 0...NTAB-1 */
763  iy=iv[j]-idum2; /* Here idum is shuffled, idum and idum2 */
764  iv[j] = *idum; /* are combined to generate output */
765  if (iy < 1) iy += IMM1;
766  if ((temp=AM*iy) > RNMX) return RNMX; /* No endpoint values */
767  else return temp;
768 }
