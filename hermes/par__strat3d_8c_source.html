98 void problem(DomainS *pDomain)
99 {
100  GridS *pGrid = pDomain->Grid;
101  int i,j,k,ks,pt,tsmode;
102  long p,q;
103  Real ScaleHg,tsmin,tsmax,tscrit,amin,amax,Hparmin,Hparmax;
104  Real *ep,*ScaleHpar,epsum,mratio,pwind,rhoaconv,etavk;
105  Real *epsilon,*uxNSH,*uyNSH,**wxNSH,**wyNSH;
106  Real rhog,h,x1,x2,x3,t,x1p,x2p,x3p,zmin,zmax,dx3_1,b;
107  long int iseed = myID_Comm_world; /* Initialize on the first call to ran2 */
108 
109  if (pDomain->Nx[2] == 1) {
110  ath_error("[par_strat3d]: par_strat3d only works for 3D problem.\n");
111  }
112 
113 #ifdef MPI_PARALLEL
114  if (pDomain->NGrid[2] > 2) {
115  ath_error(
116  "[par_strat3d]: The z-domain can not be decomposed into more than 2 grids\n");
117  }
118 #endif
119 
120 /* Initialize boxsize */
121  x1min = pGrid->MinX[0];
122  x1max = pGrid->MaxX[0];
123  Lx = x1max - x1min;
124 
125  x2min = pGrid->MinX[1];
126  x2max = pGrid->MaxX[1];
127  Ly = x2max - x2min;
128 
129  x3min = par_getd("domain1","x3min");
130  x3max = par_getd("domain1","x3max");
131  Lz = x3max - x3min;
132 
133  Lg = nghost*pGrid->dx3; /* size of the ghost zone */
134 
135  ks = pGrid->ks;
136 
137 /* Read initial conditions */
138  Omega_0 = par_getd("problem","omega");
139  qshear = par_getd_def("problem","qshear",1.5);
140  ipert = par_geti_def("problem","ipert",1);
141  vsc1 = par_getd_def("problem","vsc1",0.05); /* in unit of iso_sound (N.B.!) */
142  vsc2 = par_getd_def("problem","vsc2",0.0);
143 
144  vsc1 = vsc1 * Iso_csound;
145  vsc2 = vsc2 * Iso_csound;
146 
147  ScaleHg = Iso_csound/Omega_0;
148 
149  /* particle number */
150  Npar = (long)(par_geti("particle","parnumgrid"));
151 
152  pGrid->nparticle = Npar*npartypes;
153  for (i=0; i<npartypes; i++)
154  grproperty[i].num = Npar;
155 
156  if (pGrid->nparticle+2 > pGrid->arrsize)
157  particle_realloc(pGrid, pGrid->nparticle+2);
158 
159  ep = (Real*)calloc_1d_array(npartypes, sizeof(Real));
160  ScaleHpar = (Real*)calloc_1d_array(npartypes, sizeof(Real));
161 
162  epsilon = (Real*)calloc_1d_array(npartypes, sizeof(Real));
163  wxNSH = (Real**)calloc_2d_array(pGrid->Nx[2]+1, npartypes,sizeof(Real));
164  wyNSH = (Real**)calloc_2d_array(pGrid->Nx[2]+1, npartypes,sizeof(Real));
165  uxNSH = (Real*)calloc_1d_array(pGrid->Nx[2]+1, sizeof(Real));
166  uyNSH = (Real*)calloc_1d_array(pGrid->Nx[2]+1, sizeof(Real));
167 
168  /* particle stopping time */
169  tsmode = par_geti("particle","tsmode");
170  if (tsmode == 3) {/* fixed stopping time */
171  tsmin = par_getd("problem","tsmin"); /* in code unit */
172  tsmax = par_getd("problem","tsmax");
173  tscrit= par_getd("problem","tscrit");
174 
175  for (i=0; i<npartypes; i++) {
176  tstop0[i] = tsmin*exp(i*log(tsmax/tsmin)/MAX(npartypes-1,1.0));
177  grproperty[i].rad = tstop0[i];
178  /* use fully implicit integrator for well coupled particles */
179  if (tstop0[i] < tscrit) grproperty[i].integrator = 3;
180  }
181  }
182  else {
183  amin = par_getd("problem","amin");
184  amax = par_getd("problem","amax");
185 
186  for (i=0; i<npartypes; i++)
187  grproperty[i].rad = amin*exp(i*log(amax/amin)/MAX(npartypes-1,1.0));
188 
189  if (tsmode <= 2) {/* Epstein/General regime */
190  /* conversion factor for rhoa */
191  rhoaconv = par_getd_def("problem","rhoaconv",1.0);
192 
193  for (i=0; i<npartypes; i++)
194  grrhoa[i]=grproperty[i].rad*rhoaconv;
195  }
196 
197  if (tsmode == 1) /* General drag formula */
198  alamcoeff = par_getd("problem","alamcoeff");
199  }
200 
201  /* particle scale height */
202  Hparmax = par_getd("problem","hparmax"); /* in unit of gas scale height */
203  Hparmin = par_getd("problem","hparmin");
204  for (i=0; i<npartypes; i++)
205  ScaleHpar[i] = Hparmax*
206  exp(-i*log(Hparmax/Hparmin)/MAX(npartypes-1,1.0));
207 
208 #ifdef FEEDBACK
209  mratio = par_getd_def("problem","mratio",0.0); /* total mass fraction */
210  pwind = par_getd_def("problem","pwind",0.0); /* power law index */
211  if (mratio < 0.0)
212  ath_error("[par_strat2d]: mratio must be positive!\n");
213 
214  epsum = 0.0;
215  for (i=0; i<npartypes; i++)
216  {
217  ep[i] = pow(grproperty[i].rad,pwind); epsum += ep[i];
218  }
219 
220  for (i=0; i<npartypes; i++)
221  {
222  ep[i] = mratio*ep[i]/epsum;
223  grproperty[i].m = sqrt(2.0*PI)*ScaleHg/Lz*ep[i]*
224  pGrid->Nx[0]*pGrid->Nx[1]*pGrid->Nx[2]/Npar;
225  }
226 #else
227  mratio = 0.0;
228  for (i=0; i<npartypes; i++)
229  ep[i] = 0.0;
230 #endif
231 
232  /* NSH equilibrium */
233  for (k=pGrid->ks; k<=pGrid->ke+1; k++) {
234 
235  h = pGrid->MinX[2] + (k-pGrid->ks)*pGrid->dx3;
236  q = k - ks;
237  etavk = fabs(vsc1+vsc2*SQR(h));
238 
239  for (i=0; i<npartypes; i++) {
240  epsilon[i] = ep[i]/ScaleHpar[i]*exp(-0.5*SQR(h/ScaleHg)
241  *(SQR(1.0/ScaleHpar[i])-1.0))/erf(Lz/(sqrt(8.0)*ScaleHpar[i]*ScaleHg));
242 
243  if (tsmode != 3)
244  tstop0[i] = get_ts(pGrid,i,exp(-0.5*SQR(h/ScaleHg)),Iso_csound,etavk);
245  }
246 
247  MultiNSH(npartypes, tstop0, epsilon, etavk,
248  &uxNSH[q], &uyNSH[q], wxNSH[q], wyNSH[q]);
249  }
250 
251 /* Now set initial conditions for the gas */
252  for (k=pGrid->ks; k<=pGrid->ke; k++) {
253  for (j=pGrid->js; j<=pGrid->je; j++) {
254  for (i=pGrid->is; i<=pGrid->ie; i++) {
255  cc_pos(pGrid,i,j,k,&x1,&x2,&x3);
256 
257  rhog = exp(-0.5*SQR(x3/ScaleHg));
258  pGrid->U[k][j][i].d = rhog;
259 
260  if (ipert != 1) {/* NSH velocity */
261  pGrid->U[k][j][i].M1 = 0.5*rhog*(uxNSH[k-ks]+uxNSH[k-ks+1]);
262  pGrid->U[k][j][i].M2 = 0.5*rhog*(uyNSH[k-ks]+uyNSH[k-ks+1]);
263  } else {
264  pGrid->U[k][j][i].M1 = 0.0;
265  pGrid->U[k][j][i].M2 = 0.0;
266  }
267 
268  pGrid->U[k][j][i].M3 = 0.0;
269 #ifndef FARGO
270  pGrid->U[k][j][i].M2 -= qshear*rhog*Omega_0*x1;
271 #endif
272 
273  }}}
274 
275 /* Now set initial conditions for the particles */
276  p = 0;
277  dx3_1 = 1.0/pGrid->dx3;
278  zmin = pGrid->MinX[2];
279  zmax = pGrid->MaxX[2];
280 
281  for (q=0; q<Npar; q++) {
282 
283  for (pt=0; pt<npartypes; pt++) {
284 
285  x1p = x1min + Lx*ran2(&iseed);
286  x2p = x2min + Ly*ran2(&iseed);
287  x3p = ScaleHpar[pt]*ScaleHg*Normal(&iseed);
288  while ((x3p >= zmax) || (x3p < zmin))
289  x3p = ScaleHpar[pt]*ScaleHg*Normal(&iseed);
290 
291  pGrid->particle[p].property = pt;
292  pGrid->particle[p].x1 = x1p;
293  pGrid->particle[p].x2 = x2p;
294  pGrid->particle[p].x3 = x3p;
295 
296  if (ipert != 1) {/* NSH velocity */
297 
298  cellk(pGrid, x3p, dx3_1, &k, &b);
299  k = k-pGrid->ks; b = b - pGrid->ks;
300 
301  pGrid->particle[p].v1 = (k+1-b)*wxNSH[k][pt]+(b-k)*wxNSH[k+1][pt];
302  pGrid->particle[p].v2 = (k+1-b)*wyNSH[k][pt]+(b-k)*wyNSH[k+1][pt];
303 
304  } else {
305 
306  pGrid->particle[p].v1 = 0.0;
307  pGrid->particle[p].v2 = vsc1+vsc2*SQR(x2p);
308 
309  }
310 
311  pGrid->particle[p].v3 = 0.0;
312 #ifndef FARGO
313  pGrid->particle[p].v2 -= qshear*Omega_0*x1p;
314 #endif
315 
316  pGrid->particle[p].pos = 1; /* grid particle */
317  pGrid->particle[p].my_id = p;
318 #ifdef MPI_PARALLEL
319  pGrid->particle[p].init_id = myID_Comm_world;
320 #endif
321  p++;
322  }}
323 
324 /* enroll gravitational potential function, shearing sheet BC functions */
325  ShearingBoxPot = StratifiedDisk;
326 
327  dump_history_enroll(hst_rho_Vx_dVy, "<rho Vx dVy>");
328 
329  /* set the # of the particles in list output
330  * (by default, output 1 particle per cell)
331  */
332  nlis = par_geti_def("problem","nlis",pGrid->Nx[0]*pGrid->Nx[1]*pGrid->Nx[2]);
333 
334  /* set the number of particles to keep track of */
335  ntrack = par_geti_def("problem","ntrack",2000);
336 
337  /* set the threshold particle density */
338  dpar_thresh = par_geti_def("problem","dpar_thresh",10.0);
339 
340  /* finalize */
341  free(ep); free(ScaleHpar);
342  free(epsilon);
343  free_2d_array(wxNSH); free_2d_array(wyNSH);
344  free(uxNSH); free(uyNSH);
345 
346  return;
347 }

485 static Real StratifiedDisk(const Real x1, const Real x2, const Real x3)
486 {
487  Real z,z0,phi=0.0;
488 
489 #ifndef FARGO
490  phi -= qshear*SQR(Omega_0*x1);
491 #endif
492  /* Ensure vertical periodicity in ghost zones */
493  if(x3 > x3max)
494  z=fabs(x3-Lz);
495  else if (x3 < x3min)
496  z=fabs(x3+Lz);
497  else
498  z=fabs(x3);
499 
500  phi += 0.5*SQR(Omega_0*z);
501 
502  /* smooth the potential at domain edges */
503  z0 = 0.5*Lz+Lg;
504  phi -= SQR(Omega_0*Lg)*Lg*(2.0*z-z0)/(2.0*SQR(z0-z)); /* 3rd order sharp */
505 
506 // phi -= SQR(Omega_0*Lg)*(z/(z0-z)+log(z0/(z0-z))); /* 2nd order sharp */
507 
508  return phi;
509 }

561 void MultiNSH(int n, Real *tstop, Real *mratio, Real etavk,
562  Real *uxNSH, Real *uyNSH, Real *wxNSH, Real *wyNSH)
563 {
564  int i,j;
565  Real *Lambda1,**Lam1GamP1, **A, **B, **Tmp;
566 
567  Lambda1 = (Real*)calloc_1d_array(n, sizeof(Real)); /* Lambda^{-1} */
568  Lam1GamP1=(Real**)calloc_2d_array(n, n, sizeof(Real)); /* Lambda1*(1+Gamma) */
569  A = (Real**)calloc_2d_array(n, n, sizeof(Real));
570  B = (Real**)calloc_2d_array(n, n, sizeof(Real));
571  Tmp = (Real**)calloc_2d_array(n, n, sizeof(Real));
572 
573  /* definitions */
574  for (i=0; i<n; i++){
575  for (j=0; j<n; j++)
576  Lam1GamP1[i][j] = mratio[j];
577  Lam1GamP1[i][i] += 1.0;
578  Lambda1[i] = 1.0/(tstop[i]+1.0e-16);
579  for (j=0; j<n; j++)
580  Lam1GamP1[i][j] *= Lambda1[i];
581  }
582 
583  /* Calculate A and B */
584  MatrixMult(Lam1GamP1, Lam1GamP1, n,n,n, Tmp);
585  for (i=0; i<n; i++) Tmp[i][i] += 1.0;
586  InverseMatrix(Tmp, n, B);
587  for (i=0; i<n; i++)
588  for (j=0; j<n; j++)
589  B[i][j] *= Lambda1[j];
590  MatrixMult(Lam1GamP1, B, n,n,n, A);
591 
592  /* Obtain NSH velocities */
593  *uxNSH = 0.0; *uyNSH = 0.0;
594  for (i=0; i<n; i++){
595  wxNSH[i] = 0.0;
596  wyNSH[i] = 0.0;
597  for (j=0; j<n; j++){
598  wxNSH[i] -= B[i][j];
599  wyNSH[i] -= A[i][j];
600  }
601  wxNSH[i] *= 2.0*etavk;
602  wyNSH[i] *= etavk;
603  *uxNSH -= mratio[i]*wxNSH[i];
604  *uyNSH -= mratio[i]*wyNSH[i];
605  wyNSH[i] += etavk;
606  }
607 
608  free(Lambda1);
609  free_2d_array(A); free_2d_array(B);
610  free_2d_array(Lam1GamP1); free_2d_array(Tmp);
611 
612  return;
613 }

645 double ran2(long int *idum)
646 {
647  int j;
648  long int k;
649  static long int idum2=123456789;
650  static long int iy=0;
651  static long int iv[NTAB];
652  double temp;
653 
654  if (*idum <= 0) { /* Initialize */
655  if (-(*idum) < 1) *idum=1; /* Be sure to prevent idum = 0 */
656  else *idum = -(*idum);
657  idum2=(*idum);
658  for (j=NTAB+7;j>=0;j--) { /* Load the shuffle table (after 8 warm-ups) */
659  k=(*idum)/IQ1;
660  *idum=IA1*(*idum-k*IQ1)-k*IR1;
661  if (*idum < 0) *idum += IM1;
662  if (j < NTAB) iv[j] = *idum;
663  }
664  iy=iv[0];
665  }
666  k=(*idum)/IQ1; /* Start here when not initializing */
667  *idum=IA1*(*idum-k*IQ1)-k*IR1; /* Compute idum=(IA1*idum) % IM1 without */
668  if (*idum < 0) *idum += IM1; /* overflows by Schrage's method */
669  k=idum2/IQ2;
670  idum2=IA2*(idum2-k*IQ2)-k*IR2; /* Compute idum2=(IA2*idum) % IM2 likewise */
671  if (idum2 < 0) idum2 += IM2;
672  j=(int)(iy/NDIV); /* Will be in the range 0...NTAB-1 */
673  iy=iv[j]-idum2; /* Here idum is shuffled, idum and idum2 */
674  iv[j] = *idum; /* are combined to generate output */
675  if (iy < 1) iy += IMM1;
676  if ((temp=AM*iy) > RNMX) return RNMX; /* No endpoint values */
677  else return temp;
678 }
